129///////////////////////////////////////////////////////////////////////////
130//
131// US_Print() - Prints a string in the current window. Newlines are
132// supported.
133//
134///////////////////////////////////////////////////////////////////////////
135void
136US_Print(const char *sorg)
137{
138 char c;
139 char *sstart = strdup(sorg);
140 char *s = sstart;
141 char *se;
142 word w,h;
143
144 while (*s)
145 {
146 se = s;
147 while ((c = *se)!=0 && (c != '\n'))
148 se++;
149 *se = '\0';
150
151 USL_MeasureString(s,&w,&h);
152 px = PrintX;
153 py = PrintY;
154 USL_DrawString(s);
155
156 s = se;
157 if (c)
158 {
159 *se = c;
160 s++;
161
162 PrintX = WindowX;
163 PrintY += h;
164 }
165 else
166 PrintX += w;
167 }
168 free(sstart);
169}

198///////////////////////////////////////////////////////////////////////////
199//
200// USL_PrintInCenter() - Prints a string in the center of the given rect
201//
202///////////////////////////////////////////////////////////////////////////
203void
204USL_PrintInCenter(const char *s,Rect r)
205{
206 word w,h,
207 rw,rh;
208
209 USL_MeasureString(s,&w,&h);
210 rw = r.lr.x - r.ul.x;
211 rh = r.lr.y - r.ul.y;
212
213 px = r.ul.x + ((rw - w) / 2);
214 py = r.ul.y + ((rh - h) / 2);
215 USL_DrawString(s);
216}

342///////////////////////////////////////////////////////////////////////////
343//
344// US_DrawWindow() - Draws a frame and sets the current window parms
345//
346///////////////////////////////////////////////////////////////////////////
347void
348US_DrawWindow(word x,word y,word w,word h)
349{
350
351 word i,
352 sx,sy,sw,sh;
353
354 WindowX = x * 8;
355 WindowY = y * 8;
356 WindowW = w * 8;
357 WindowH = h * 8;
358
359 PrintX = WindowX;
360 PrintY = WindowY;
361
362 sx = (x - 1) * 8;
363 sy = (y - 1) * 8;
364 sw = (w + 1) * 8;
365 sh = (h + 1) * 8;
366
367 US_ClearWindow();
368
369 VWB_DrawTile8(sx,sy,0),VWB_DrawTile8(sx,sy + sh,5);
370 for (i = sx + 8;i <= sx + sw - 8;i += 8)
371 VWB_DrawTile8(i,sy,1),VWB_DrawTile8(i,sy + sh,6);
372 VWB_DrawTile8(i,sy,2),VWB_DrawTile8(i,sy + sh,7);
373
374 for (i = sy + 8;i <= sy + sh - 8;i += 8)
375 VWB_DrawTile8(sx,i,3),VWB_DrawTile8(sx + sw,i,4);
376}

421static void
422USL_XORICursor(int x,int y,const char *s,word cursor)
423{
424 static boolean status; // VGA doesn't XOR...
425 char buf[MaxString];
426 int temp;
427 word w,h;
428
429 strcpy(buf,s);
430 buf[cursor] = '\0';
431 USL_MeasureString(buf,&w,&h);
432
433 px = x + w - 1;
434 py = y;
435 if (status^=1)
436 USL_DrawString("\x80");
437 else
438 {
439 temp = fontcolor;
440 fontcolor = backcolor;
441 USL_DrawString("\x80");
442 fontcolor = temp;
443 }
444}

446char USL_RotateChar(char ch, int dir)
447{
448 static const char charSet[] = " ABCDEFGHIJKLMNOPQRSTUVWXYZ.,-!?0123456789";
449 const int numChars = sizeof(charSet) / sizeof(char) - 1;
450 int i;
451 for(i = 0; i < numChars; i++)
452 {
453 if(ch == charSet[i]) break;
454 }
455
456 if(i == numChars) i = 0;
457
458 i += dir;
459 if(i < 0) i = numChars - 1;
460 else if(i >= numChars) i = 0;
461 return charSet[i];
462}

464///////////////////////////////////////////////////////////////////////////
465//
466// US_LineInput() - Gets a line of user input at (x,y), the string defaults
467// to whatever is pointed at by def. Input is restricted to maxchars
468// chars or maxwidth pixels wide. If the user hits escape (and escok is
469// true), nothing is copied into buf, and false is returned. If the
470// user hits return, the current string is copied into buf, and true is
471// returned
472//
473///////////////////////////////////////////////////////////////////////////
474boolean
475US_LineInput(int x,int y,char *buf,const char *def,boolean escok,
476 int maxchars,int maxwidth)
477{
478 boolean redraw,
479 cursorvis,cursormoved,
480 done,result, checkkey;
481 ScanCode sc;
482 char c;
483 char s[MaxString],olds[MaxString];
484 int cursor,len;
485 word i,
486 w,h,
487 temp;
488 longword curtime, lasttime, lastdirtime, lastbuttontime, lastdirmovetime;
489 ControlInfo ci;
490 Direction lastdir = dir_None;
491
492 if (def)
493 strcpy(s,def);
494 else
495 *s = '\0';
496 *olds = '\0';
497 cursor = (int) strlen(s);
498 cursormoved = redraw = true;
499
500 cursorvis = done = false;
501 lasttime = lastdirtime = lastdirmovetime = GetTimeCount();
502 lastbuttontime = lasttime + TickBase / 4; // 250 ms => first button press accepted after 500 ms
503 LastASCII = key_None;
504 LastScan = sc_None;
505
506 while (!done)
507 {
508 ReadAnyControl(&ci);
509
510 if (cursorvis)
511 USL_XORICursor(x,y,s,cursor);
512
513 sc = LastScan;
514 LastScan = sc_None;
515 c = LastASCII;
516 LastASCII = key_None;
517
518 checkkey = true;
519 curtime = GetTimeCount();
520
521 // After each direction change accept the next change after 250 ms and then everz 125 ms
522 if(ci.dir != lastdir || curtime - lastdirtime > TickBase / 4 && curtime - lastdirmovetime > TickBase / 8)
523 {
524 if(ci.dir != lastdir)
525 {
526 lastdir = ci.dir;
527 lastdirtime = curtime;
528 }
529 lastdirmovetime = curtime;
530
531 switch(ci.dir)
532 {
533 case dir_West:
534 if(cursor)
535 {
536 // Remove trailing whitespace if cursor is at end of string
537 if(s[cursor] == ' ' && s[cursor + 1] == 0)
538 s[cursor] = 0;
539 cursor--;
540 }
541 cursormoved = true;
542 checkkey = false;
543 break;
544 case dir_East:
545 if(cursor >= MaxString - 1) break;
546
547 if(!s[cursor])
548 {
549 USL_MeasureString(s,&w,&h);
550 if(len >= maxchars || maxwidth && w >= maxwidth) break;
551
552 s[cursor] = ' ';
553 s[cursor + 1] = 0;
554 }
555 cursor++;
556 cursormoved = true;
557 checkkey = false;
558 break;
559
560 case dir_North:
561 if(!s[cursor])
562 {
563 USL_MeasureString(s,&w,&h);
564 if(len >= maxchars || maxwidth && w >= maxwidth) break;
565 s[cursor + 1] = 0;
566 }
567 s[cursor] = USL_RotateChar(s[cursor], 1);
568 redraw = true;
569 checkkey = false;
570 break;
571
572 case dir_South:
573 if(!s[cursor])
574 {
575 USL_MeasureString(s,&w,&h);
576 if(len >= maxchars || maxwidth && w >= maxwidth) break;
577 s[cursor + 1] = 0;
578 }
579 s[cursor] = USL_RotateChar(s[cursor], -1);
580 redraw = true;
581 checkkey = false;
582 break;
583 }
584 }
585
586 if((int)(curtime - lastbuttontime) > TickBase / 4) // 250 ms
587 {
588 if(ci.button0) // acts as return
589 {
590 strcpy(buf,s);
591 done = true;
592 result = true;
593 checkkey = false;
594 }
595 if(ci.button1 && escok) // acts as escape
596 {
597 done = true;
598 result = false;
599 checkkey = false;
600 }
601 if(ci.button2) // acts as backspace
602 {
603 lastbuttontime = curtime;
604 if(cursor)
605 {
606 strcpy(s + cursor - 1,s + cursor);
607 cursor--;
608 redraw = true;
609 }
610 cursormoved = true;
611 checkkey = false;
612 }
613 }
614
615 if(checkkey)
616 {
617 switch (sc)
618 {
619 case sc_LeftArrow:
620 if (cursor)
621 cursor--;
622 c = key_None;
623 cursormoved = true;
624 break;
625 case sc_RightArrow:
626 if (s[cursor])
627 cursor++;
628 c = key_None;
629 cursormoved = true;
630 break;
631 case sc_Home:
632 cursor = 0;
633 c = key_None;
634 cursormoved = true;
635 break;
636 case sc_End:
637 cursor = (int) strlen(s);
638 c = key_None;
639 cursormoved = true;
640 break;
641
642 case sc_Return:
643 strcpy(buf,s);
644 done = true;
645 result = true;
646 c = key_None;
647 break;
648 case sc_Escape:
649 if (escok)
650 {
651 done = true;
652 result = false;
653 }
654 c = key_None;
655 break;
656
657 case sc_BackSpace:
658 if (cursor)
659 {
660 strcpy(s + cursor - 1,s + cursor);
661 cursor--;
662 redraw = true;
663 }
664 c = key_None;
665 cursormoved = true;
666 break;
667 case sc_Delete:
668 if (s[cursor])
669 {
670 strcpy(s + cursor,s + cursor + 1);
671 redraw = true;
672 }
673 c = key_None;
674 cursormoved = true;
675 break;
676
677 case SDLK_KP5: //0x4c: // Keypad 5 // TODO: hmmm...
678 case sc_UpArrow:
679 case sc_DownArrow:
680 case sc_PgUp:
681 case sc_PgDn:
682 case sc_Insert:
683 c = key_None;
684 break;
685 }
686
687 if (c)
688 {
689 len = (int) strlen(s);
690 USL_MeasureString(s,&w,&h);
691
692 if(isprint(c) && (len < MaxString - 1) && ((!maxchars) || (len < maxchars))
693 && ((!maxwidth) || (w < maxwidth)))
694 {
695 for (i = len + 1;i > cursor;i--)
696 s[i] = s[i - 1];
697 s[cursor++] = c;
698 redraw = true;
699 }
700 }
701 }
702
703 if (redraw)
704 {
705 px = x;
706 py = y;
707 temp = fontcolor;
708 fontcolor = backcolor;
709 USL_DrawString(olds);
710 fontcolor = (byte) temp;
711 strcpy(olds,s);
712
713 px = x;
714 py = y;
715 USL_DrawString(s);
716
717 redraw = false;
718 }
719
720 if (cursormoved)
721 {
722 cursorvis = false;
723 lasttime = curtime - TickBase;
724
725 cursormoved = false;
726 }
727 if (curtime - lasttime > TickBase / 2) // 500 ms
728 {
729 lasttime = curtime;
730
731 cursorvis ^= true;
732 }
733 else SDL_Delay(5);
734 if (cursorvis)
735 USL_XORICursor(x,y,s,cursor);
736
737 VW_UpdateScreen();
738 }
739
740 if (cursorvis)
741 USL_XORICursor(x,y,s,cursor);
742 if (!result)
743 {
744 px = x;
745 py = y;
746 USL_DrawString(olds);
747 }
748 VW_UpdateScreen();
749
750 IN_ClearKeysDown();
751 return(result);
752}

754///////////////////////////////////////////////////////////////////////////
755//
756// US_InitRndT - Initializes the pseudo random number generator.
757// If randomize is true, the seed will be initialized depending on the
758// current time
759//
760///////////////////////////////////////////////////////////////////////////
761void US_InitRndT(int randomize)
762{
763 if(randomize)
764 rndindex = (SDL_GetTicks() >> 4) & 0xff;
765 else
766 rndindex = 0;
767}

769///////////////////////////////////////////////////////////////////////////
770//
771// US_RndT - Returns the next 8-bit pseudo random number
772//
773///////////////////////////////////////////////////////////////////////////
774int US_RndT()
775{
776 rndindex = (rndindex+1)&0xff;
777 return rndtable[rndindex];
778}
