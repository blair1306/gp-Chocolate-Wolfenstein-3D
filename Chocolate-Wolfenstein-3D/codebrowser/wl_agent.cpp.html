93/*
94======================
95=
96= CheckWeaponChange
97=
98= Keys 1-4 change weapons
99=
100======================
101*/
102
103void CheckWeaponChange (void)
104{
105 int newWeapon = -1;
106
107 if (!gamestate.ammo) // must use knife with no ammo
108 return;
109
110 if(buttonstate[bt_nextweapon] && !buttonheld[bt_nextweapon])
111 {
112 newWeapon = gamestate.weapon + 1;
113 if(newWeapon > gamestate.bestweapon) newWeapon = 0;
114 }
115 else if(buttonstate[bt_prevweapon] && !buttonheld[bt_prevweapon])
116 {
117 newWeapon = gamestate.weapon - 1;
118 if(newWeapon < 0) newWeapon = gamestate.bestweapon;
119 }
120 else
121 {
122 for(int i = wp_knife; i <= gamestate.bestweapon; i++)
123 {
124 if (buttonstate[bt_readyknife + i - wp_knife])
125 {
126 newWeapon = i;
127 break;
128 }
129 }
130 }
131
132 if(newWeapon != -1)
133 {
134 gamestate.weapon = gamestate.chosenweapon = (weapontype) newWeapon;
135 DrawWeapon();
136 }
137}

140/*
141=======================
142=
143= ControlMovement
144=
145= Takes controlx,controly, and buttonstate[bt_strafe]
146=
147= Changes the player's angle and position
148=
149= There is an angle hack because when going 70 fps, the roundoff becomes
150= significant
151=
152=======================
153*/
154
155void ControlMovement (objtype *ob)
156{
157 int32_t oldx,oldy;
158 int angle;
159 int angleunits;
160
161 thrustspeed = 0;
162
163 oldx = player->x;
164 oldy = player->y;
165
166 if(buttonstate[bt_strafeleft])
167 {
168 angle = ob->angle + ANGLES/4;
169 if(angle >= ANGLES)
170 angle -= ANGLES;
171 if(buttonstate[bt_run])
172 Thrust(angle, RUNMOVE * MOVESCALE * tics);
173 else
174 Thrust(angle, BASEMOVE * MOVESCALE * tics);
175 }
176
177 if(buttonstate[bt_straferight])
178 {
179 angle = ob->angle - ANGLES/4;
180 if(angle < 0)
181 angle += ANGLES;
182 if(buttonstate[bt_run])
183 Thrust(angle, RUNMOVE * MOVESCALE * tics );
184 else
185 Thrust(angle, BASEMOVE * MOVESCALE * tics);
186 }
187
188 //
189 // side to side move
190 //
191 if (buttonstate[bt_strafe])
192 {
193 //
194 // strafing
195 //
196 //
197 if (controlx > 0)
198 {
199 angle = ob->angle - ANGLES/4;
200 if (angle < 0)
201 angle += ANGLES;
202 Thrust (angle,controlx*MOVESCALE); // move to left
203 }
204 else if (controlx < 0)
205 {
206 angle = ob->angle + ANGLES/4;
207 if (angle >= ANGLES)
208 angle -= ANGLES;
209 Thrust (angle,-controlx*MOVESCALE); // move to right
210 }
211 }
212 else
213 {
214 //
215 // not strafing
216 //
217 anglefrac += controlx;
218 angleunits = anglefrac/ANGLESCALE;
219 anglefrac -= angleunits*ANGLESCALE;
220 ob->angle -= angleunits;
221
222 if (ob->angle >= ANGLES)
223 ob->angle -= ANGLES;
224 if (ob->angle < 0)
225 ob->angle += ANGLES;
226
227 }
228
229 //
230 // forward/backwards move
231 //
232 if (controly < 0)
233 {
234 Thrust (ob->angle,-controly*MOVESCALE); // move forwards
235 }
236 else if (controly > 0)
237 {
238 angle = ob->angle + ANGLES/2;
239 if (angle >= ANGLES)
240 angle -= ANGLES;
241 Thrust (angle,controly*BACKMOVESCALE); // move backwards
242 }
243
244 if (gamestate.victoryflag) // watching the BJ actor
245 return;
246}

362static void LatchNumber (int x, int y, unsigned width, int32_t number)
363{
364 unsigned length,c;
365 char str[20];
366
367 ltoa (number,str,10);
368
369 length = (unsigned) strlen (str);
370
371 while (length<width)
372 {
373 StatusDrawPic (x,y,N_BLANKPIC);
374 x++;
375 width--;
376 }
377
378 c = length <= width ? 0 : length-width;
379
380 while (c<length)
381 {
382 StatusDrawPic (x,y,str[c]-'0'+ N_0PIC);
383 x++;
384 c++;
385 }
386}

817/*
818===================
819=
820= TryMove
821=
822= returns true if move ok
823= debug: use pointers to optimize
824===================
825*/
826
827boolean TryMove (objtype *ob)
828{
829 int xl,yl,xh,yh,x,y;
830 objtype *check;
831 int32_t deltax,deltay;
832
833 xl = (ob->x-PLAYERSIZE) >>TILESHIFT;
834 yl = (ob->y-PLAYERSIZE) >>TILESHIFT;
835
836 xh = (ob->x+PLAYERSIZE) >>TILESHIFT;
837 yh = (ob->y+PLAYERSIZE) >>TILESHIFT;
838
839#define PUSHWALLMINDIST PLAYERSIZE
840
841 //
842 // check for solid walls
843 //
844 for (y=yl;y<=yh;y++)
845 {
846 for (x=xl;x<=xh;x++)
847 {
848 check = actorat[x][y];
849 if (check && !ISPOINTER(check))
850 {
851 if(tilemap[x][y]==64 && x==pwallx && y==pwally) // back of moving pushwall?
852 {
853 switch(pwalldir)
854 {
855 case di_north:
856 if(ob->y-PUSHWALLMINDIST<=(pwally<<TILESHIFT)+((63-pwallpos)<<10))
857 return false;
858 break;
859 case di_west:
860 if(ob->x-PUSHWALLMINDIST<=(pwallx<<TILESHIFT)+((63-pwallpos)<<10))
861 return false;
862 break;
863 case di_east:
864 if(ob->x+PUSHWALLMINDIST>=(pwallx<<TILESHIFT)+(pwallpos<<10))
865 return false;
866 break;
867 case di_south:
868 if(ob->y+PUSHWALLMINDIST>=(pwally<<TILESHIFT)+(pwallpos<<10))
869 return false;
870 break;
871 }
872 }
873 else return false;
874 }
875 }
876 }
877
878 //
879 // check for actors
880 //
881 if (yl>0)
882 yl--;
883 if (yh<MAPSIZE-1)
884 yh++;
885 if (xl>0)
886 xl--;
887 if (xh<MAPSIZE-1)
888 xh++;
889
890 for (y=yl;y<=yh;y++)
891 {
892 for (x=xl;x<=xh;x++)
893 {
894 check = actorat[x][y];
895 if (ISPOINTER(check) && check != player && (check->flags & FL_SHOOTABLE) )
896 {
897 deltax = ob->x - check->x;
898 if (deltax < -MINACTORDIST || deltax > MINACTORDIST)
899 continue;
900 deltay = ob->y - check->y;
901 if (deltay < -MINACTORDIST || deltay > MINACTORDIST)
902 continue;
903
904 return false;
905 }
906 }
907 }
908
909 return true;
910}

920
921void ClipMove (objtype *ob, int32_t xmove, int32_t ymove)
922{
923 int32_t basex,basey;
924
925 basex = ob->x;
926 basey = ob->y;
927
928 ob->x = basex+xmove;
929 ob->y = basey+ymove;
930 if (TryMove (ob))
931 return;
932
933#ifndef REMDEBUG
934 if (noclip && ob->x > 2*TILEGLOBAL && ob->y > 2*TILEGLOBAL
935 && ob->x < (((int32_t)(mapwidth-1))<<TILESHIFT)
936 && ob->y < (((int32_t)(mapheight-1))<<TILESHIFT) )
937 return; // walk through walls
938#endif
939
940 if (!SD_SoundPlaying())
941 SD_PlaySound (HITWALLSND);
942
943 ob->x = basex+xmove;
944 ob->y = basey;
945 if (TryMove (ob))
946 return;
947
948 ob->x = basex;
949 ob->y = basey+ymove;
950 if (TryMove (ob))
951 return;
952
953 ob->x = basex;
954 ob->y = basey;
955}

984// For player movement in demos exactly as in the original Wolf3D v1.4 source code
985static fixed FixedByFracOrig(fixed a, fixed b)
986{
987 int sign = 0;
988 if(b == 65536)
989 b = 65535;
990 else
991 if(b == -65536) b = 65535, sign = 1;
992 else
993 if(b < 0) b = (-b), sign = 1;
994
995 if(a < 0)
996 {
997 a = -a;
998 sign = !sign;
999 }
1000 fixed res = (fixed)(((int64_t) a * b) >> 16);
1001
1002 if(sign)
1003 res = -res;
1004 return res;
1005}

1007void Thrust (int angle, int32_t speed)
1008{
1009 int32_t xmove,ymove;
1010 unsigned offset;
1011
1012
1013 //
1014 // ZERO FUNNY COUNTER IF MOVED!
1015 //
1016#ifdef SPEAR
1017 if (speed)
1018 funnyticount = 0;
1019#endif
1020
1021 thrustspeed += speed;
1022 //
1023 // moving bounds speed
1024 //
1025 if (speed >= MINDIST*2)
1026 speed = MINDIST*2-1;
1027
1028 xmove = DEMOCHOOSE_ORIG_SDL(
1029 FixedByFracOrig(speed, costable[angle]),
1030 FixedMul(speed,costable[angle]));
1031 ymove = DEMOCHOOSE_ORIG_SDL(
1032 -FixedByFracOrig(speed, sintable[angle]),
1033 -FixedMul(speed,sintable[angle]));
1034
1035 ClipMove(player,xmove,ymove);
1036
1037 player->tilex = (short)(player->x >> TILESHIFT); // scale to tile values
1038 player->tiley = (short)(player->y >> TILESHIFT);

1245void GunAttack (objtype *ob)
1246{
1247 objtype *check,*closest,*oldclosest;

1250 int32_t viewdist;
1251
1252 switch (gamestate.weapon)

1258 SD_PlaySound (ATKMACHINEGUNSND);

1262 break;

1264
1265 madenoise = true;
1266
1267 //
1268 // find potential targets
1269 //
1270 viewdist = 0x7fffffffl;
1271 closest = NULL;
1272

1276

1278 {
1279 if ((check->flags & FL_SHOOTABLE) && (check->flags & FL_VISABLE)
1280 && abs(check->viewx-centerx) < shootdelta)
1281 {
1282 if (check->transx < viewdist)
1283 {
1284 viewdist = check->transx;
1285 closest = check;
1286 }
1287 }
1288 }
1289
1290 if (closest == oldclosest)
1291 return; // no more targets, all missed
1292
1293 //
1294 // trace a line from player to enemey
1295 //
1296 if (CheckLine(closest))
1297 break;

1300 //
1301 // hit something
1302 //

1304 dy = ABS(closest->tiley - player->tiley);

1313 return;
