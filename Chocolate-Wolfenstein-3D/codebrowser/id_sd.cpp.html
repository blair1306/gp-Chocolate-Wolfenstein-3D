161void inline SDL_DoFX()
162{
163 if(pcSound)
164 {
165 if(*pcSound!=pcLastSample)
166 {
167 pcLastSample=*pcSound;
168
169 if(pcLastSample)
170 SDL_turnOnPCSpeaker(pcLastSample*60);
171 else
172 SDL_turnOffPCSpeaker();
173 }
174 pcSound++;
175 pcLengthLeft--;
176 if(!pcLengthLeft)
177 {
178 pcSound=0;
179 SoundNumber=(soundnames)0;
180 SoundPriority=0;
181 SDL_turnOffPCSpeaker();
182 }
183 }
184
185 // [adlib sound stuff removed...]
186}

226 if(pcSound)

286static void
287SDL_SetTimer0(word speed)
288{
289#ifndef TPROF // If using Borland's profiling, don't screw with the timer
290// _asm pushfd
291 _asm cli
292
293 outp(0x43,0x36); // Change timer 0
294 outp(0x40,(byte)speed);
295 outp(0x40,speed >> 8);
296 // Kludge to handle special case for digitized PC sounds
297 if (TimerDivisor == (1192030 / (TickBase * 100)))
298 TimerDivisor = (1192030 / (TickBase * 10));
299 else
300 TimerDivisor = speed;
301
302// _asm popfd
303 _asm sti
304#else
305 TimerDivisor = 0x10000;
306#endif
307}

322static void
323SDL_SetTimerSpeed(void)
324{
325 word rate;
326 void (_interrupt *isr)(void);
327
328 if ((DigiMode == sds_PC) && DigiPlaying)
329 {
330 rate = TickBase * 100;
331 isr = SDL_t0ExtremeAsmService;
332 }
333 else if ((MusicMode == smm_AdLib) || ((DigiMode == sds_SoundSource) && DigiPlaying) )
334 {
335 rate = TickBase * 10;
336 isr = SDL_t0FastAsmService;
337 }
338 else
339 {
340 rate = TickBase * 2;
341 isr = SDL_t0SlowAsmService;
342 }
343
344 if (rate != TimerRate)
345 {
346 _dos_setvect(8,isr);
347 SDL_SetIntsPerSec(rate);
348 TimerRate = rate;
349 }
350}

527Sint16 GetSample(float csample, byte *samples, int size)
528{
529 float s0=0, s1=0, s2=0;
530 int cursample = (int) csample;
531 float sf = csample - (float) cursample;
532
533 if(cursample-1 >= 0) s0 = (float) (samples[cursample-1] - 128);
534 s1 = (float) (samples[cursample] - 128);
535 if(cursample+1 < size) s2 = (float) (samples[cursample+1] - 128);
536
537 float val = s0*sf*(sf-1)/2 - s1*(sf*sf-1) + s2*(sf+1)*sf/2;
538 int32_t intval = (int32_t) (val * 256);
539 if(intval < -32768) intval = -32768;
540 else if(intval > 32767) intval = 32767;
541 return (Sint16) intval;
542}

544void SD_PrepareSound(int which)
545{
546 if(DigiList == NULL)
547 Quit("SD_PrepareSound(%i): DigiList not initialized!\n", which);
548
549 int page = DigiList[which].startpage;
550 int size = DigiList[which].length;
551
552 byte *origsamples = PM_GetSound(page);
553 if(origsamples + size >= PM_GetEnd())
554 Quit("SD_PrepareSound(%i): Sound reaches out of page file!\n", which);
555
556 int destsamples = (int) ((float) size * (float) param_samplerate
557 / (float) ORIGSAMPLERATE);
558
559 byte *wavebuffer = (byte *) malloc(sizeof(headchunk) + sizeof(wavechunk)
560 + destsamples * 2); // dest are 16-bit samples
561 if(wavebuffer == NULL)
562 Quit("Unable to allocate wave buffer for sound %i!\n", which);
563
564 headchunk head = {{'R','I','F','F'}, 0, {'W','A','V','E'},
565 {'f','m','t',' '}, 0x10, 0x0001, 1, param_samplerate, param_samplerate*2, 2, 16};
566 wavechunk dhead = {{'d', 'a', 't', 'a'}, destsamples*2};
567 head.filelenminus8 = sizeof(head) + destsamples*2; // (sizeof(dhead)-8 = 0)
568 memcpy(wavebuffer, &head, sizeof(head));
569 memcpy(wavebuffer+sizeof(head), &dhead, sizeof(dhead));
570
571 // alignment is correct, as wavebuffer comes from malloc
572 // and sizeof(headchunk) % 4 == 0 and sizeof(wavechunk) % 4 == 0
573 Sint16 *newsamples = (Sint16 *)(void *) (wavebuffer + sizeof(headchunk)
574 + sizeof(wavechunk));
575 float cursample = 0.F;
576 float samplestep = (float) ORIGSAMPLERATE / (float) param_samplerate;
577 for(int i=0; i<destsamples; i++, cursample+=samplestep)
578 {
579 newsamples[i] = GetSample((float)size * (float)i / (float)destsamples,
580 origsamples, size);
581 }
582 SoundBuffers[which] = wavebuffer;
583
584 SoundChunks[which] = Mix_LoadWAV_RW(SDL_RWFromMem(wavebuffer,
585 sizeof(headchunk) + sizeof(wavechunk) + destsamples * 2), 1);
586}

588int SD_PlayDigitized(word which,int leftpos,int rightpos)
589{
590 if (!DigiMode)
591 return 0;
592
593 if (which >= NumDigi)
594 Quit("SD_PlayDigitized: bad sound number %i", which);
595
596 int channel = SD_GetChannelForDigi(which);
597 SD_SetPosition(channel, leftpos,rightpos);
598
599 DigiPlaying = true;
600
601 Mix_Chunk *sample = SoundChunks[which];
602 if(sample == NULL)
603 {
604 printf("SoundChunks[%i] is NULL!\n", which);
605 return 0;
606 }
607
608 if(Mix_PlayChannel(channel, sample, 0) == -1)
609 {
610 printf("Unable to play sound: %s\n", Mix_GetError());
611 return 0;
612 }
613
614 return channel;
615}

651void
652SDL_SetupDigi(void)
653{
654 // Correct padding enforced by PM_Startup()
655 word *soundInfoPage = (word *) (void *) PM_GetPage(ChunksInFile-1);
656 NumDigi = (word) PM_GetPageSize(ChunksInFile - 1) / 4;
657
658 DigiList = (digiinfo *) malloc(NumDigi * sizeof(digiinfo));
659 int i;
660 for(i = 0; i < NumDigi; i++)
661 {
662 // Calculate the size of the digi from the sizes of the pages between
663 // the start page and the start page of the next sound
664
665 DigiList[i].startpage = soundInfoPage[i * 2];
666 if((int) DigiList[i].startpage >= ChunksInFile - 1)
667 {
668 NumDigi = i;
669 break;
670 }
671
672 int lastPage;
673 if(i < NumDigi - 1)
674 {
675 lastPage = soundInfoPage[i * 2 + 2];
676 if(lastPage == 0 || lastPage + PMSoundStart > ChunksInFile - 1) lastPage = ChunksInFile - 1;
677 else lastPage += PMSoundStart;
678 }
679 else lastPage = ChunksInFile - 1;
680
681 int size = 0;
682 for(int page = PMSoundStart + DigiList[i].startpage; page < lastPage; page++)
683 size += PM_GetPageSize(page);
684
685 // Don't include padding of sound info page, if padding was added
686 if(lastPage == ChunksInFile - 1 && PMSoundInfoPagePadded) size--;
687
688 // Patch lower 16-bit of size with size from sound info page.
689 // The original VSWAP contains padding which is included in the page size,
690 // but not included in the 16-bit size. So we use the more precise value.
691 if((size & 0xffff0000) != 0 && (size & 0xffff) < soundInfoPage[i * 2 + 1])
692 size -= 0x10000;
693 size = (size & 0xffff0000) | soundInfoPage[i * 2 + 1];
694
695 DigiList[i].length = size;
696 }
697
698 for(i = 0; i < LASTSOUND; i++)
699 {
700 DigiMap[i] = -1;
701 DigiChannel[i] = -1;
702 }
703}

881///////////////////////////////////////////////////////////////////////////
882//
883// SD_SetSoundMode() - Sets which sound hardware to use for sound effects
884//
885///////////////////////////////////////////////////////////////////////////
886boolean
887SD_SetSoundMode(SDMode mode)
888{
889 boolean result = false;
890 word tableoffset;
891
892 SD_StopSound();
893
894 if ((mode == sdm_AdLib) && !AdLibPresent)
895 mode = sdm_PC;
896
897 switch (mode)
898 {
899 case sdm_Off:
900 tableoffset = STARTADLIBSOUNDS;
901 result = true;
902 break;
903 case sdm_PC:
904 tableoffset = STARTPCSOUNDS;
905 result = true;
906 break;
907 case sdm_AdLib:
908 tableoffset = STARTADLIBSOUNDS;
909 if (AdLibPresent)
910 result = true;
911 break;
912 default:

961int numreadysamples = 0;
962byte *curAlSound = 0;
963byte *curAlSoundPtr = 0;
964longword curAlLengthLeft = 0;
965int soundTimeCounter = 5;
966int samplesPerMusicTick;
967
968void SDL_IMFMusicPlayer(void *udata, Uint8 *stream, int len)
969{
970 int stereolen = len>>1;
971 int sampleslen = stereolen>>1;
972 INT16 *stream16 = (INT16 *) (void *) stream; // expect correct alignment
973
974 while(1)
975 {
976 if(numreadysamples)
977 {
978 if(numreadysamples<sampleslen)
979 {
980 YM3812UpdateOne(0, stream16, numreadysamples);
981 stream16 += numreadysamples*2;
982 sampleslen -= numreadysamples;
983 }
984 else
985 {
986 YM3812UpdateOne(0, stream16, sampleslen);
987 numreadysamples -= sampleslen;
988 return;
989 }
990 }
991 soundTimeCounter--;
992 if(!soundTimeCounter)
993 {
994 soundTimeCounter = 5;
995 if(curAlSound != alSound)
996 {
997 curAlSound = curAlSoundPtr = alSound;
998 curAlLengthLeft = alLengthLeft;
999 }
1000 if(curAlSound)
1001 {
1002 if(*curAlSoundPtr)
1003 {
1004 alOut(alFreqL, *curAlSoundPtr);
1005 alOut(alFreqH, alBlock);
1006 }
1007 else alOut(alFreqH, 0);
1008 curAlSoundPtr++;
1009 curAlLengthLeft--;
1010 if(!curAlLengthLeft)
1011 {
1012 curAlSound = alSound = 0;
1013 SoundNumber = (soundnames) 0;
1014 SoundPriority = 0;
1015 alOut(alFreqH, 0);
1016 }
1017 }
1018 }
1019 if(sqActive)
1020 {
1021 do
1022 {
1023 if(sqHackTime > alTimeCount) break;
1024 sqHackTime = alTimeCount + *(sqHackPtr+1);
1025 alOut(*(byte *) sqHackPtr, *(((byte *) sqHackPtr)+1));
1026 sqHackPtr += 2;
1027 sqHackLen -= 4;
1028 }
1029 while(sqHackLen>0);
1030 alTimeCount++;
1031 if(!sqHackLen)
1032 {
1033 sqHackPtr = sqHack;
1034 sqHackLen = sqHackSeqLen;
1035 sqHackTime = 0;
1036 alTimeCount = 0;
1037 }
1038 }
1039 numreadysamples = samplesPerMusicTick;
1040 }
1041}

1075 for(i=1;i<0xf6;i++)
1076 YM3812Write(0,i,0);

1350
1351 if (MusicMode == smm_AdLib)
1352 {
1353 int32_t chunkLen = CA_CacheAudioChunk(chunk);
1354 sqHack = (word *)(void *) audiosegs[chunk]; // alignment is correct
1355 if(*sqHack == 0) sqHackLen = sqHackSeqLen = chunkLen;
1356 else sqHackLen = sqHackSeqLen = *sqHack++;
1357 sqHackPtr = sqHack;
1358
1359 if(startoffs >= sqHackLen)
1360 {
1361 Quit("SD_StartMusic: Illegal startoffs provided!");
1362 }
1363
1364 // fast forward to correct position
1365 // (needed to reconstruct the instruments)
1366
1367 for(int i = 0; i < startoffs; i += 2)
1368 {
1369 byte reg = *(byte *)sqHackPtr;
1370 byte val = *(((byte *)sqHackPtr) + 1);
1371 if(reg >= 0xb1 && reg <= 0xb8) val &= 0xdf; // disable play note flag
1372 else if(reg == 0xbd) val &= 0xe0; // disable drum flags
1373
1374 alOut(reg,val);
1375 sqHackPtr += 2;
1376 sqHackLen -= 4;
1377 }
1378 sqHackTime = 0;
1379 alTimeCount = 0;
1380
1381 SD_MusicOn();
1382 }
