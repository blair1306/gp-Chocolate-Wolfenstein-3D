142void SpawnStatic (int tilex, int tiley, int type)
143{
144 laststatobj->shapenum = statinfo[type].picnum;
145 laststatobj->tilex = tilex;
146 laststatobj->tiley = tiley;
147 laststatobj->visspot = &spotvis[tilex][tiley];
148
149 switch (statinfo[type].type)
150 {
151 case block:
152 actorat[tilex][tiley] = (objtype *) 64; // consider it a blocking tile
153 case none:
154 laststatobj->flags = 0;
155 break;
156
157 case bo_cross:
158 case bo_chalice:
159 case bo_bible:
160 case bo_crown:
161 case bo_fullheal:
162 if (!loadedgame)
163 gamestate.treasuretotal++;
164
165 case bo_firstaid:
166 case bo_key1:
167 case bo_key2:
168 case bo_key3:
169 case bo_key4:
170 case bo_clip:
171 case bo_25clip:
172 case bo_machinegun:
173 case bo_chaingun:
174 case bo_food:
175 case bo_alpo:
176 case bo_gibs:
177 case bo_spear:
178 laststatobj->flags = FL_BONUS;
179 laststatobj->itemnumber = statinfo[type].type;
180 break;
181 }
182
183 laststatobj->flags |= statinfo[type].specialFlags;
184
185 laststatobj++;
186
187 if (laststatobj == &statobjlist[MAXSTATS])
188 Quit ("Too many static objects!\n");
189}

192/*
193===============
194=
195= PlaceItemType
196=
197= Called during game play to drop actors' items. It finds the proper
198= item number based on the item type (bo_???). If there are no free item
199= spots, nothing is done.
200=
201===============
202*/
203
204void PlaceItemType (int itemtype, int tilex, int tiley)
205{
206 int type;
207 statobj_t *spot;
208
209 //
210 // find the item number
211 //
212 for (type=0; ; type++)
213 {
214 if (statinfo[type].picnum == -1) // end of list
215 Quit ("PlaceItemType: couldn't find type!");
216 if (statinfo[type].type == itemtype)
217 break;
218 }
219
220 //
221 // find a spot in statobjlist to put it in
222 //
223 for (spot=&statobjlist[0]; ; spot++)
224 {
225 if (spot==laststatobj)
226 {
227 if (spot == &statobjlist[MAXSTATS])
228 return; // no free spots
229 laststatobj++; // space at end
230 break;
231 }
232
233 if (spot->shapenum == -1) // -1 is a free spot
234 break;
235 }
236 //
237 // place it
238 //
239 spot->shapenum = statinfo[type].picnum;
240 spot->tilex = tilex;
241 spot->tiley = tiley;
242 spot->visspot = &spotvis[tilex][tiley];
243 spot->flags = FL_BONUS | statinfo[type].specialFlags;
244 spot->itemnumber = statinfo[type].type;
245}

249/*
250=============================================================================
251
252 DOORS
253
254doorobjlist[] holds most of the information for the doors
255
256doorposition[] holds the amount the door is open, ranging from 0 to 0xffff
257 this is directly accessed by AsmRefresh during rendering
258
259The number of doors is limited to 64 because a spot in tilemap holds the
260 door number in the low 6 bits, with the high bit meaning a door center
261 and bit 6 meaning a door side tile
262
263Open doors conect two areas, so sounds will travel between them and sight
264 will be checked when the player is in a connected area.
265
266Areaconnect is incremented/decremented by each door. If >0 they connect
267
268Every time a door opens or closes the areabyplayer matrix gets recalculated.
269 An area is true if it connects with the player's current spor.
270
271=============================================================================
272*/
273
274#define DOORWIDTH 0x7800
275#define OPENTICS 300

280word doorposition[MAXDOORS]; // leading edge of door 0=closed
281 // 0xffff = fully open

288/*
289==============
290=
291= ConnectAreas
292=
293= Scans outward from playerarea, marking all connected areas
294=
295==============
296*/
297
298void RecursiveConnect (int areanumber)
299{
300 int i;
301
302 for (i=0;i<NUMAREAS;i++)
303 {
304 if (areaconnect[areanumber][i] && !areabyplayer[i])
305 {
306 areabyplayer[i] = true;
307 RecursiveConnect (i);
308 }
309 }
310}

314{

316 areabyplayer[player->areanumber] = true;
317 RecursiveConnect (player->areanumber);

349===============
350=
351= SpawnDoor
352=
353===============

355
356void SpawnDoor (int tilex, int tiley, boolean vertical, int lock)
357{
358 word *map;
359
360 if (doornum==MAXDOORS)
361 Quit ("64+ doors on level!");
362
363 doorposition[doornum] = 0; // doors start out fully closed
364 lastdoorobj->tilex = tilex;
365 lastdoorobj->tiley = tiley;
366 lastdoorobj->vertical = vertical;
367 lastdoorobj->lock = lock;
368 lastdoorobj->action = dr_closed;
369
370 actorat[tilex][tiley] = (objtype *)(uintptr_t)(doornum | 0x80); // consider it a solid wall
371
372 //
373 // make the door tile a special tile, and mark the adjacent tiles
374 // for door sides
375 //
376 tilemap[tilex][tiley] = doornum | 0x80;
377 map = mapsegs[0] + (tiley<<mapshift) +tilex;
378 if (vertical)
379 {
380 *map = *(map-1); // set area number
381 tilemap[tilex][tiley-1] |= 0x40;
382 tilemap[tilex][tiley+1] |= 0x40;
383 }
384 else
385 {
386 *map = *(map-mapwidth); // set area number
387 tilemap[tilex-1][tiley] |= 0x40;
388 tilemap[tilex+1][tiley] |= 0x40;
389 }
390
391 doornum++;
392 lastdoorobj++;
393}

414/*
415=====================
416=
417= CloseDoor
418=
419=====================
420*/
421
422void CloseDoor (int door)
423{
424 int tilex,tiley,area;
425 objtype *check;
426
427 //
428 // don't close on anything solid
429 //
430 tilex = doorobjlist[door].tilex;
431 tiley = doorobjlist[door].tiley;
432
433 if (actorat[tilex][tiley])
434 return;
435
436 if (player->tilex == tilex && player->tiley == tiley)
437 return;
438
439 if (doorobjlist[door].vertical)
440 {
441 if ( player->tiley == tiley )
442 {
443 if ( ((player->x+MINDIST) >>TILESHIFT) == tilex )
444 return;
445 if ( ((player->x-MINDIST) >>TILESHIFT) == tilex )
446 return;
447 }
448 check = actorat[tilex-1][tiley];
449 if (ISPOINTER(check) && ((check->x+MINDIST) >> TILESHIFT) == tilex )
450 return;
451 check = actorat[tilex+1][tiley];
452 if (ISPOINTER(check) && ((check->x-MINDIST) >> TILESHIFT) == tilex )
453 return;
454 }
455 else if (!doorobjlist[door].vertical)
456 {
457 if (player->tilex == tilex)
458 {
459 if ( ((player->y+MINDIST) >>TILESHIFT) == tiley )
460 return;
461 if ( ((player->y-MINDIST) >>TILESHIFT) == tiley )
462 return;
463 }
464 check = actorat[tilex][tiley-1];
465 if (ISPOINTER(check) && ((check->y+MINDIST) >> TILESHIFT) == tiley )
466 return;
467 check = actorat[tilex][tiley+1];
468 if (ISPOINTER(check) && ((check->y-MINDIST) >> TILESHIFT) == tiley )
469 return;
470 }
471
472
473 //
474 // play door sound if in a connected area
475 //
476 area = *(mapsegs[0] + (doorobjlist[door].tiley<<mapshift)
477 +doorobjlist[door].tilex)-AREATILE;
478 if (areabyplayer[area])
479 {
480 PlaySoundLocTile(CLOSEDOORSND,doorobjlist[door].tilex,doorobjlist[door].tiley); // JAB
481 }
482
483 doorobjlist[door].action = dr_closing;
484 //
485 // make the door space solid
486 //
487 actorat[tilex][tiley] = (objtype *)(uintptr_t)(door | 0x80);
488}

492/*
493=====================
494=
495= OperateDoor
496=
497= The player wants to change the door's direction
498=
499=====================
500*/
501
502void OperateDoor (int door)
503{
504 int lock;
505
506 lock = doorobjlist[door].lock;
507 if (lock >= dr_lock1 && lock <= dr_lock4)
508 {
509 if ( ! (gamestate.keys & (1 << (lock-dr_lock1) ) ) )
510 {
511 SD_PlaySound (NOWAYSND); // locked
512 return;
513 }
514 }
515
516 switch (doorobjlist[door].action)
517 {
518 case dr_closed:
519 case dr_closing:
520 OpenDoor (door);
521 break;
522 case dr_open:
523 case dr_opening:
524 CloseDoor (door);
525 break;
526 }
527}

550/*
551===============
552=
553= DoorOpening
554=
555===============
556*/
557
558void DoorOpening (int door)
559{
560 unsigned area1,area2;
561 word *map;
562 int32_t position;
563
564 position = doorposition[door];
565 if (!position)
566 {
567 //
568 // door is just starting to open, so connect the areas
569 //
570 map = mapsegs[0] + (doorobjlist[door].tiley<<mapshift)
571 +doorobjlist[door].tilex;
572
573 if (doorobjlist[door].vertical)
574 {
575 area1 = *(map+1);
576 area2 = *(map-1);
577 }
578 else
579 {
580 area1 = *(map-mapwidth);
581 area2 = *(map+mapwidth);
582 }
583 area1 -= AREATILE;
584 area2 -= AREATILE;
585
586 if (area1 < NUMAREAS && area2 < NUMAREAS)
587 {
588 areaconnect[area1][area2]++;
589 areaconnect[area2][area1]++;
590
591 if (player->areanumber < NUMAREAS)
592 ConnectAreas ();
593
594 if (areabyplayer[area1])
595 PlaySoundLocTile(OPENDOORSND,doorobjlist[door].tilex,doorobjlist[door].tiley); // JAB
596 }
597 }
598
599 //
600 // slide the door by an adaptive amount
601 //
602 position += tics<<10;
603 if (position >= 0xffff)
604 {
605 //
606 // door is all the way open
607 //
608 position = 0xffff;
609 doorobjlist[door].ticcount = 0;
610 doorobjlist[door].action = dr_open;
611 actorat[doorobjlist[door].tilex][doorobjlist[door].tiley] = 0;
612 }
613
614 doorposition[door] = (word) position;
615}

618/*
619===============
620=
621= DoorClosing
622=
623===============
624*/
625
626void DoorClosing (int door)
627{
628 unsigned area1,area2;
629 word *map;
630 int32_t position;
631 int tilex,tiley;
632
633 tilex = doorobjlist[door].tilex;
634 tiley = doorobjlist[door].tiley;
635
636 if ( ((int)(uintptr_t)actorat[tilex][tiley] != (door | 0x80))
637 || (player->tilex == tilex && player->tiley == tiley) )
638 { // something got inside the door
639 OpenDoor (door);
640 return;
641 };
642
643 position = doorposition[door];
644
645 //
646 // slide the door by an adaptive amount
647 //
648 position -= tics<<10;
649 if (position <= 0)
650 {
651 //
652 // door is closed all the way, so disconnect the areas
653 //
654 position = 0;
655
656 doorobjlist[door].action = dr_closed;
657
658 map = mapsegs[0] + (doorobjlist[door].tiley<<mapshift) + doorobjlist[door].tilex;
659
660 if (doorobjlist[door].vertical)
661 {
662 area1 = *(map+1);
663 area2 = *(map-1);
664 }
665 else
666 {
667 area1 = *(map-mapwidth);
668 area2 = *(map+mapwidth);
669 }
670 area1 -= AREATILE;
671 area2 -= AREATILE;
672
673 if (area1 < NUMAREAS && area2 < NUMAREAS)
674 {
675 areaconnect[area1][area2]--;
676 areaconnect[area2][area1]--;
677
678 if (player->areanumber < NUMAREAS)
679 ConnectAreas ();
680 }
681 }
682
683 doorposition[door] = (word) position;
684}

738int dirs[4][2]={{0,-1},{1,0},{0,1},{-1,0}};

740/*
741===============
742=
743= PushWall
744=
745===============
746*/
747
748void PushWall (int checkx, int checky, int dir)
749{
750 int oldtile, dx, dy;
751
752 if (pwallstate)
753 return;
754
755 oldtile = tilemap[checkx][checky];
756 if (!oldtile)
757 return;
758
759 dx = dirs[dir][0];
760 dy = dirs[dir][1];
761
762 if (actorat[checkx+dx][checky+dy])
763 {
764 SD_PlaySound (NOWAYSND);
765 return;
766 }
767 actorat[checkx+dx][checky+dy] = (objtype *)(uintptr_t) (tilemap[checkx+dx][checky+dy] = oldtile);
768
769 gamestate.secretcount++;
770 pwallx = checkx;
771 pwally = checky;
772 pwalldir = dir;
773 pwallstate = 1;
774 pwallpos = 0;
775 pwalltile = tilemap[pwallx][pwally];
776 tilemap[pwallx][pwally] = 64;
777 tilemap[pwallx+dx][pwally+dy] = 64;
778 *(mapsegs[1]+(pwally<<mapshift)+pwallx) = 0; // remove P tile info
779 *(mapsegs[0]+(pwally<<mapshift)+pwallx) = *(mapsegs[0]+(player->tiley<<mapshift)+player->tilex); // set correct floorcode (BrotherTank's fix)
780
781 SD_PlaySound (PUSHWALLSND);
782}

786/*
787=================
788=
789= MovePWalls
790=
791=================
792*/
793
794void MovePWalls (void)
795{
796 int oldblock,oldtile;
797
798 if (!pwallstate)
799 return;
800
801 oldblock = pwallstate/128;
802
803 pwallstate += (word)tics;
804
805 if (pwallstate/128 != oldblock)
806 {
807 // block crossed into a new block
808 oldtile = pwalltile;
809
810 //
811 // the tile can now be walked into
812 //
813 tilemap[pwallx][pwally] = 0;
814 actorat[pwallx][pwally] = 0;
815 *(mapsegs[0]+(pwally<<mapshift)+pwallx) = player->areanumber+AREATILE;
816
817 int dx=dirs[pwalldir][0], dy=dirs[pwalldir][1];
818 //
819 // see if it should be pushed farther
820 //
821 if (pwallstate>=256) // only move two tiles fix
822 {
823 //
824 // the block has been pushed two tiles
825 //
826 pwallstate = 0;
827 tilemap[pwallx+dx][pwally+dy] = oldtile;
828 return;
829 }
830 else
831 {
832 int xl,yl,xh,yh;
833 xl = (player->x-PLAYERSIZE) >> TILESHIFT;
834 yl = (player->y-PLAYERSIZE) >> TILESHIFT;
835 xh = (player->x+PLAYERSIZE) >> TILESHIFT;
836 yh = (player->y+PLAYERSIZE) >> TILESHIFT;
837
838 pwallx += dx;
839 pwally += dy;
840
841 if (actorat[pwallx+dx][pwally+dy]
842 || xl<=pwallx+dx && pwallx+dx<=xh && yl<=pwally+dy && pwally+dy<=yh)
843 {
844 pwallstate = 0;
845 tilemap[pwallx][pwally] = oldtile;
846 return;
847 }
848 actorat[pwallx+dx][pwally+dy] = (objtype *)(uintptr_t) (tilemap[pwallx+dx][pwally+dy] = oldtile);
849 tilemap[pwallx+dx][pwally+dy] = 64;
850 }
851 }
852
853 pwallpos = (pwallstate/2)&63;
854}
