124void
125SetSoundLoc(fixed gx,fixed gy)
126{
127 fixed xt,yt;
128 int x,y;
129
130//
131// translate point to view centered coordinates
132//
133 gx -= viewx;
134 gy -= viewy;
135
136//
137// calculate newx
138//
139 xt = FixedMul(gx,viewcos);
140 yt = FixedMul(gy,viewsin);
141 x = (xt - yt) >> TILESHIFT;
142
143//
144// calculate newy
145//
146 xt = FixedMul(gx,viewsin);
147 yt = FixedMul(gy,viewcos);
148 y = (yt + xt) >> TILESHIFT;
149
150 if (y >= ATABLEMAX)
151 y = ATABLEMAX - 1;
152 else if (y <= -ATABLEMAX)
153 y = -ATABLEMAX;
154 if (x < 0)
155 x = -x;
156 if (x >= ATABLEMAX)
157 x = ATABLEMAX - 1;
158 leftchannel = lefttable[x][y + ATABLEMAX];
159 rightchannel = righttable[x][y + ATABLEMAX];
160
161}

163/*
164==========================
165=
166= SetSoundLocGlobal - Sets up globalsoundx & globalsoundy and then calls
167= UpdateSoundLoc() to transform that into relative channel volumes. Those
168= values are then passed to the Sound Manager so that they'll be used for
169= the next sound played (if possible).
170=
171= JAB
172=
173==========================
174*/
175void PlaySoundLocGlobal(word s,fixed gx,fixed gy)
176{
177 SetSoundLoc(gx, gy);
178 SD_PositionSound(leftchannel, rightchannel);
179
180 int channel = SD_PlaySound((soundnames) s);
181 if(channel)
182 {
183 channelSoundPos[channel - 1].globalsoundx = gx;
184 channelSoundPos[channel - 1].globalsoundy = gy;
185 channelSoundPos[channel - 1].valid = 1;
186 }
187}

628void SetupGameLevel (void)
629{
630 int x,y;

636 {
637 gamestate.TimeCount

639 = gamestate.killtotal

644 = pwallstate = pwallpos = facetimes = 0;

646 killerobj = NULL;
647 }
648
649 if (demoplayback || demorecord)
650 US_InitRndT (false);
651 else
652 US_InitRndT (true);
653
654//
655// load the level
656//
657 CA_CacheMap (gamestate.mapon+10*gamestate.episode);
658 mapon-=gamestate.episode*10;
659
660
661//
662// copy the wall data to a data segment array
663//
664 memset (tilemap,0,sizeof(tilemap));
665 memset (actorat,0,sizeof(actorat));
666 map = mapsegs[0];
667 for (y=0;y<mapheight;y++)
668 {
669 for (x=0;x<mapwidth;x++)
670 {
671 tile = *map++;
672 if (tile<AREATILE)
673 {
674 // solid wall
675 tilemap[x][y] = (byte) tile;
676 actorat[x][y] = (objtype *)(uintptr_t) tile;
677 }
678 else
679 {
680 // area floor
681 tilemap[x][y] = 0;
682 actorat[x][y] = 0;
683 }
684 }
685 }
686
687//
688// spawn doors
689//
690 InitActorList (); // start spawning things with a clean slate
691 InitDoorList ();
692 InitStaticList ();
693
694 map = mapsegs[0];
695 for (y=0;y<mapheight;y++)
696 {
697 for (x=0;x<mapwidth;x++)
698 {
699 tile = *map++;
700 if (tile >= 90 && tile <= 101)
701 {
702 // door
703 switch (tile)
704 {
705 case 90:
706 case 92:
707 case 94:
708 case 96:
709 case 98:
710 case 100:
711 SpawnDoor (x,y,1,(tile-90)/2);
712 break;
713 case 91:
714 case 93:
715 case 95:
716 case 97:
717 case 99:
718 case 101:
719 SpawnDoor (x,y,0,(tile-91)/2);
720 break;
721 }
722 }
723 }
724 }
725
726//
727// spawn actors
728//
729 ScanInfoPlane ();
730
731
732//
733// take out the ambush markers
734//
735 map = mapsegs[0];
736 for (y=0;y<mapheight;y++)
737 {
738 for (x=0;x<mapwidth;x++)
739 {
740 tile = *map++;
741 if (tile == AMBUSHTILE)
742 {
743 tilemap[x][y] = 0;
744 if ( (unsigned)(uintptr_t)actorat[x][y] == AMBUSHTILE)
745 actorat[x][y] = NULL;
746
747 if (*map >= AREATILE)
748 tile = *map;
749 if (*(map-1-mapwidth) >= AREATILE)
750 tile = *(map-1-mapwidth);
751 if (*(map-1+mapwidth) >= AREATILE)
752 tile = *(map-1+mapwidth);
753 if ( *(map-2) >= AREATILE)
754 tile = *(map-2);
755
756 *(map-1) = tile;
757 }
758 }
759 }
760
761
762//
763// have the caching manager load and purge stuff to make sure all marks
764// are in memory
765//
766 CA_LoadAllSounds ();
767}

795

803 {
804 VWB_BarScaledCoord(0, 0, sw, yl - px, bordercol); // upper side
805 VWB_BarScaledCoord(0, yl + vh + px, sw, yl - px, bordercol); // lower side
806 }
807
808 if(xl != 0)
809 {
810 // Paint game view border lines
811 VWB_BarScaledCoord(xl - px, yl - px, vw + px, px, 0); // upper border
812 VWB_BarScaledCoord(xl, yl + vh, vw + px, px, bordercol - 2); // lower border

816 }

1208
1209void Died (void)
1210{
1211 float fangle;
1212 int32_t dx,dy;
1213 int iangle,curangle,clockwise,counter,change;
1214
1215 if (screenfaded)
1216 {
1217 ThreeDRefresh ();
1218 VW_FadeIn ();
1219 }
1220
1221 gamestate.weapon = (weapontype) -1; // take away weapon
1222 SD_PlaySound (PLAYERDEATHSND);
1223
1224 //
1225 // swing around to face attacker
1226 //
1227 if(killerobj)
1228 {
1229 dx = killerobj->x - player->x;
1230 dy = player->y - killerobj->y;
1231
1232 fangle = (float) atan2((float) dy, (float) dx); // returns -pi to pi
1233 if (fangle<0)
1234 fangle = (float) (M_PI*2+fangle);
1235
1236 iangle = (int) (fangle/(M_PI*2)*ANGLES);
1237 }
1238 else
1239 {
1240 iangle = player->angle + ANGLES / 2;
1241 if(iangle >= ANGLES) iangle -= ANGLES;
1242 }
1243
1244 if (player->angle > iangle)
1245 {
1246 counter = player->angle - iangle;
1247 clockwise = ANGLES-player->angle + iangle;
1248 }
1249 else
1250 {
1251 clockwise = iangle - player->angle;
1252 counter = player->angle + ANGLES-iangle;
1253 }
1254
1255 curangle = player->angle;
1256
1257 if (clockwise<counter)
1258 {
1259 //
1260 // rotate clockwise
1261 //
1262 if (curangle>iangle)
1263 curangle -= ANGLES;
1264 do
1265 {
1266 change = tics*DEATHROTATE;
1267 if (curangle + change > iangle)
1268 change = iangle-curangle;
1269
1270 curangle += change;
1271 player->angle += change;
1272 if (player->angle >= ANGLES)
1273 player->angle -= ANGLES;
1274
1275 ThreeDRefresh ();
1276 CalcTics ();
1277 } while (curangle != iangle);
1278 }
1279 else
1280 {
1281 //
1282 // rotate counterclockwise
1283 //
1284 if (curangle<iangle)
1285 curangle += ANGLES;
1286 do
1287 {
1288 change = -(int)tics*DEATHROTATE;
1289 if (curangle + change < iangle)
1290 change = iangle-curangle;
1291
1292 curangle += change;
1293 player->angle += change;
1294 if (player->angle < 0)
1295 player->angle += ANGLES;
1296
1297 ThreeDRefresh ();
1298 CalcTics ();
1299 } while (curangle != iangle);
1300 }
1301
1302 //
1303 // fade to red
1304 //
1305 FinishPaletteShifts ();
1306
1307 if(usedoublebuffering) VH_UpdateScreen();
1308
1309 VL_BarScaledCoord (viewscreenx,viewscreeny,viewwidth,viewheight,4);
1310
1311 IN_ClearKeysDown ();
1312
1313 FizzleFade(screenBuffer,viewscreenx,viewscreeny,viewwidth,viewheight,70,false);
1314
1315 IN_UserInput(100);
1316 SD_WaitSoundDone ();
1317 ClearMemory();
1318
1319 gamestate.lives--;
1320
1321 if (gamestate.lives > -1)
1322 {
1323 gamestate.health = 100;
1324 gamestate.weapon = gamestate.bestweapon
1325 = gamestate.chosenweapon = wp_pistol;
1326 gamestate.ammo = STARTAMMO;
1327 gamestate.keys = 0;
1328 pwallstate = pwallpos = 0;
1329 gamestate.attackframe = gamestate.attackcount =
1330 gamestate.weaponframe = 0;

1333 {
1334 DrawKeys ();
1335 DrawWeapon ();
