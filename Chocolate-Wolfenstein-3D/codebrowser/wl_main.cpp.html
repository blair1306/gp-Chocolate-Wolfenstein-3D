115void ReadConfig(void)
116{
117 SDMode sd;
118 SMMode sm;
119 SDSMode sds;
120
121 char configpath[300];
122
123
124 if(configdir[0])
125 snprintf(configpath, sizeof(configpath), "%s/%s", configdir, configname);
126 else
127 strcpy(configpath, configname);
128
129 const int file = open(configpath, O_RDONLY | O_BINARY);
130 if (file != -1)
131 {
132 //
133 // valid config file
134 //
135 word tmp;
136 read(file,&tmp,sizeof(tmp));
137 if(tmp!=0xfefa)
138 {
139 close(file);
140 goto noconfig;
141 }
142 read(file,Scores,sizeof(HighScore) * MaxScores);
143
144 read(file,&sd,sizeof(sd));
145 read(file,&sm,sizeof(sm));
146 read(file,&sds,sizeof(sds));
147
148 read(file,&mouseenabled,sizeof(mouseenabled));
149 read(file,&joystickenabled,sizeof(joystickenabled));
150 boolean dummyJoypadEnabled;
151 read(file,&dummyJoypadEnabled,sizeof(dummyJoypadEnabled));
152 boolean dummyJoystickProgressive;
153 read(file,&dummyJoystickProgressive,sizeof(dummyJoystickProgressive));
154 int dummyJoystickPort = 0;
155 read(file,&dummyJoystickPort,sizeof(dummyJoystickPort));
156
157 read(file,dirscan,sizeof(dirscan));
158 read(file,buttonscan,sizeof(buttonscan));
159 read(file,buttonmouse,sizeof(buttonmouse));
160 read(file,buttonjoy,sizeof(buttonjoy));
161
162 read(file,&viewsize,sizeof(viewsize));
163 read(file,&mouseadjustment,sizeof(mouseadjustment));
164
165 close(file);
166
167 if ((sd == sdm_AdLib || sm == smm_AdLib) && !AdLibPresent
168 && !SoundBlasterPresent)
169 {
170 sd = sdm_PC;
171 sm = smm_Off;
172 }
173
174 if ((sds == sds_SoundBlaster && !SoundBlasterPresent))
175 sds = sds_Off;
176
177 // make sure values are correct
178
179 if(mouseenabled) mouseenabled=true;
180 if(joystickenabled) joystickenabled=true;
181
182 if (!MousePresent)
183 mouseenabled = false;
184 if (!IN_JoyPresent())
185 joystickenabled = false;
186
187 if(mouseadjustment<0) mouseadjustment=0;
188 else if(mouseadjustment>9) mouseadjustment=9;
189
190 if(viewsize<4) viewsize=4;
191 else if(viewsize>21) viewsize=21;
192
193 MainMenu[6].active=1;
194 MainItems.curpos=0;
195 }
196 else
197 {
198 //
199 // no config file, so select by hardware
200 //
201noconfig:
202 if (SoundBlasterPresent || AdLibPresent)
203 {
204 sd = sdm_AdLib;
205 sm = smm_AdLib;
206 }
207 else
208 {
209 sd = sdm_PC;
210 sm = smm_Off;
211 }
212
213 if (SoundBlasterPresent)
214 sds = sds_SoundBlaster;
215 else
216 sds = sds_Off;
217
218 if (MousePresent)
219 mouseenabled = true;
220
221 if (IN_JoyPresent())
222 joystickenabled = true;
223
224 viewsize = 19; // start with a good size
225 mouseadjustment=5;
226 }
227
228 SD_SetMusicMode (sm);
229 SD_SetSoundMode (sd);
230 SD_SetDigiDevice (sds);
231}

324int32_t DoChecksum(byte *source,unsigned size,int32_t checksum)
325{
326 unsigned i;
327
328 for (i=0;i<size-1;i++)
329 checksum += source[i]^source[i+1];
330

635

637 gamestate.lives = 1;
638 gamestate.weapon =
639 gamestate.chosenweapon =
640 gamestate.bestweapon = wp_pistol;
641 gamestate.ammo = 8;
642 }
643
644 return true;
645}

672/*
673==================
674=
675= BuildTables
676=
677= Calculates:
678=
679= scale projection constant
680= sintable/costable overlapping fractional tables
681=
682==================
683*/
684
685const float radtoint = (float)(FINEANGLES/2/PI);
686
687void BuildTables (void)
688{
689 //
690 // calculate fine tangents
691 //
692
693 int i;
694 for(i=0;i<FINEANGLES/8;i++)
695 {
696 double tang=tan((i+0.5)/radtoint);
697 finetangent[i]=(int32_t)(tang*GLOBAL1);
698 finetangent[FINEANGLES/4-1-i]=(int32_t)((1/tang)*GLOBAL1);
699 }
700
701 //
702 // costable overlays sintable with a quarter phase shift
703 // ANGLES is assumed to be divisable by four
704 //
705
706 float angle=0;
707 float anglestep=(float)(PI/2/ANGLEQUAD);
708 for(i=0; i<ANGLEQUAD; i++)
709 {
710 fixed value=(int32_t)(GLOBAL1*sin(angle));
711 sintable[i]=sintable[i+ANGLES]=sintable[ANGLES/2-i]=value;
712 sintable[ANGLES-i]=sintable[ANGLES/2+i]=-value;
713 angle+=anglestep;
714 }
715 sintable[ANGLEQUAD] = 65536;
716 sintable[3*ANGLEQUAD] = -65536;
717
718}

723/*
724====================
725=
726= CalcProjection
727=
728= Uses focallength
729=
730====================
731*/
732
733void CalcProjection (int32_t focal)
734{
735 int i;
736 int intang;
737 float angle;
738 double tang;
739 int halfview;
740 double facedist;
741
742 focallength = focal;
743 facedist = focal+MINDIST;
744 halfview = viewwidth/2; // half view in pixels
745
746 //
747 // calculate scale value for vertical height calculations
748 // and sprite x calculations
749 //
750 scale = (fixed) (halfview*facedist/(VIEWGLOBAL/2));
751
752 //
753 // divide heightnumerator by a posts distance to get the posts height for
754 // the heightbuffer. The pixel height is height>>2
755 //
756 heightnumerator = (TILEGLOBAL*scale)>>6;
757
758 //
759 // calculate the angle offset from view angle of each pixel's ray
760 //
761
762 for (i=0;i<halfview;i++)
763 {
764 // start 1/2 pixel over, so viewangle bisects two middle pixels
765 tang = (int32_t)i*VIEWGLOBAL/viewwidth/facedist;
766 angle = (float) atan(tang);
767 intang = (int) (angle*radtoint);
768 pixelangle[halfview-1-i] = intang;
769 pixelangle[halfview+i] = -intang;
770 }
771}

1329 viewscreenx = (screenWidth-viewwidth) / 2;
1330 viewscreeny = (screenHeight-scaleFactor*STATUSLINES-viewheight)/2;

1335// calculate trace angles and projection constants
1336//
1337 CalcProjection (FOCALLENGTH);
1338
1339 return true;
