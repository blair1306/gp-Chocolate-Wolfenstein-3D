122//
123// transform actor
124//
125void TransformActor (objtype *ob)
126{
127 fixed gx,gy,gxt,gyt,nx,ny;
128
129//
130// translate point to view centered coordinates
131//
132 gx = ob->x-viewx;
133 gy = ob->y-viewy;
134
135//
136// calculate newx
137//
138 gxt = FixedMul(gx,viewcos);
139 gyt = FixedMul(gy,viewsin);
140 nx = gxt-gyt-ACTORSIZE; // fudge the shape forward a bit, because
141 // the midpoint could put parts of the shape
142 // into an adjacent wall
143
144//
145// calculate newy
146//
147 gxt = FixedMul(gx,viewsin);
148 gyt = FixedMul(gy,viewcos);
149 ny = gyt+gxt;
150
151//
152// calculate perspective ratio
153//
154 ob->transx = nx;
155 ob->transy = ny;
156
157 if (nx<MINDIST) // too close, don't overflow the divide
158 {
159 ob->viewheight = 0;
160 return;
161 }
162
163 ob->viewx = (word)(centerx + ny*scale/nx);
164
165//
166// calculate height (heightnumerator/(nx>>8))
167//
168 ob->viewheight = (word)(heightnumerator/(nx>>8));
169}

173/*
174========================
175=
176= TransformTile
177=
178= Takes paramaters:
179= tx,ty : tile the object is centered in
180=
181= globals:
182= viewx,viewy : point of view
183= viewcos,viewsin : sin/cos of viewangle
184= scale : conversion from global value to screen value
185=
186= sets:
187= screenx,transx,transy,screenheight: projected edge location and size
188=
189= Returns true if the tile is withing getting distance
190=
191========================
192*/
193
194boolean TransformTile (int tx, int ty, short *dispx, short *dispheight)
195{
196 fixed gx,gy,gxt,gyt,nx,ny;
197
198//
199// translate point to view centered coordinates
200//
201 gx = ((int32_t)tx<<TILESHIFT)+0x8000-viewx;
202 gy = ((int32_t)ty<<TILESHIFT)+0x8000-viewy;
203
204//
205// calculate newx
206//
207 gxt = FixedMul(gx,viewcos);
208 gyt = FixedMul(gy,viewsin);
209 nx = gxt-gyt-0x2000; // 0x2000 is size of object
210
211//
212// calculate newy
213//
214 gxt = FixedMul(gx,viewsin);
215 gyt = FixedMul(gy,viewcos);
216 ny = gyt+gxt;
217
218
219//
220// calculate height / perspective ratio
221//
222 if (nx<MINDIST) // too close, don't overflow the divide
223 *dispheight = 0;
224 else
225 {
226 *dispx = (short)(centerx + ny*scale/nx);
227 *dispheight = (short)(heightnumerator/(nx>>8));
228 }
229
230//
231// see if it should be grabbed
232//
233 if (nx<TILEGLOBAL && ny>-TILEGLOBAL/2 && ny<TILEGLOBAL/2)
234 return true;
235 else
236 return false;
237}

241/*
242====================
243=
244= CalcHeight
245=
246= Calculates the height of xintercept,yintercept from viewx,viewy
247=
248====================
249*/
250
251int CalcHeight()
252{
253 fixed z = FixedMul(xintercept - viewx, viewcos) - FixedMul(yintercept - viewy, viewsin);
254
255
256 if(z < MINDIST)
257 z = MINDIST;
258
259 int height = heightnumerator / (z >> 8);
260
261 if(height < min_wallheight)
262 min_wallheight = height;
263
264 return height;
265}

276
277byte *postsource;
278int postx;
279
280void ScalePost()
281{
282
283 int ywcount, yoffs, yw, yd, yendoffs;
284 byte col;
285
286
287
288 ywcount = yd = wallheight[postx] >> 3;
289 if(yd <= 0) yd = 100;
290
291 yoffs = (viewheight / 2 - ywcount) * vbufPitch;
292 if(yoffs < 0) yoffs = 0;
293 yoffs += postx;
294
295 yendoffs = viewheight / 2 + ywcount - 1;
296 yw=TEXTURESIZE-1;
297
298 while(yendoffs >= viewheight)
299 {
300 ywcount -= TEXTURESIZE/2;
301 while(ywcount <= 0)
302 {
303 ywcount += yd;
304 yw--;
305 }
306 yendoffs--;
307 }
308 if(yw < 0) return;
309
310 col = postsource[yw];
311
312 yendoffs = yendoffs * vbufPitch + postx;
313 while(yoffs <= yendoffs)
314 {
315 vbuf[yendoffs] = col;
316 ywcount -= TEXTURESIZE/2;
317 if(ywcount <= 0)
318 {
319 do
320 {
321 ywcount += yd;
322 yw--;
323 }
324 while(ywcount <= 0);
325 if(yw < 0) break;
326 col = postsource[yw];
327 }
328 yendoffs -= vbufPitch;
329 }
330}

335 vbufPitch = pitch;
336 ScalePost();

342= HitVertWall
343=
344= tilehit bit 7 is 0, because it's not a door tile
345= if bit 6 is 1 and the adjacent tile is a door tile, use door side pic

350void HitVertWall (void)
351{
352 int wallpic;
353 int texture;
354
355 texture = ((yintercept+texdelta)>>TEXTUREFROMFIXEDSHIFT)&TEXTUREMASK;
356 if (xtilestep == -1)
357 {
358 texture = TEXTUREMASK-texture;
359 xintercept += TILEGLOBAL;
360 }
361
362 if(lastside==1 && lastintercept==xtile && lasttilehit==tilehit && !(lasttilehit & 0x40))
363 {
364 if((pixx&3) && texture == lasttexture)
365 {
366 ScalePost();
367 postx = pixx;
368 wallheight[pixx] = wallheight[pixx-1];
369 return;
370 }
371 ScalePost();
372 wallheight[pixx] = CalcHeight();
373 postsource+=texture-lasttexture;
374 postx=pixx;
375 lasttexture=texture;
376 return;
377 }
378
379 if(lastside!=-1) ScalePost();
380
381 lastside=1;
382 lastintercept=xtile;
383 lasttilehit=tilehit;
384 lasttexture=texture;
385 wallheight[pixx] = CalcHeight();
386 postx = pixx;
387
388 if (tilehit & 0x40)
389 { // check for adjacent doors
390 ytile = (short)(yintercept>>TILESHIFT);
391 if ( tilemap[xtile-xtilestep][ytile]&0x80 )
392 wallpic = DOORWALL+3;
393 else
394 wallpic = vertwall[tilehit & ~0x40];
395 }
396 else
397 wallpic = vertwall[tilehit];
398
399 postsource = PM_GetTexture(wallpic) + texture;
400}

640=====================
641=
642= CalcRotate
643=
644=====================
645*/
646
647int CalcRotate (objtype *ob)
648{
649 int angle, viewangle;
650
651 // this isn't exactly correct, as it should vary by a trig value,
652 // but it is close enough with only eight rotations
653
654 viewangle = player->angle + (centerx - ob->viewx)/8;
655
656 if (ob->obclass == rocketobj || ob->obclass == hrocketobj)
657 angle = (viewangle-180) - ob->angle;
658 else
659 angle = (viewangle-180) - dirangle[ob->dir];
660
661 angle+=ANGLES/16;
662 while (angle>=ANGLES)
663 angle-=ANGLES;
664 while (angle<0)
665 angle+=ANGLES;
666
667 if (ob->state->rotate == 2) // 2 rotation pain frame
668 return 0; // pain with shooting frame bugfix
669
670 return angle/(ANGLES/8);
671}

673void ScaleShape (int xcenter, int shapenum, unsigned height, uint32_t flags)
674{
675 t_compshape *shape;
676 unsigned scale,pixheight;
677 unsigned starty,endy;
678 word *cmdptr;
679 byte *cline;
680 byte *line;
681 byte *vmem;
682 int actx,i,upperedge;
683 short newstart;
684 int scrstarty,screndy,lpix,rpix,pixcnt,ycnt;
685 unsigned j;
686 byte col;
687
688
689
690 shape = (t_compshape *) PM_GetSprite(shapenum);
691
692 scale=height>>3; // low three bits are fractional
693 if(!scale) return; // too close or far away
694
695 pixheight=scale*SPRITESCALEFACTOR;
696 actx=xcenter-scale;
697 upperedge=viewheight/2-scale;
698
699 cmdptr=(word *) shape->dataofs;
700
701 for(i=shape->leftpix,pixcnt=i*pixheight,rpix=(pixcnt>>6)+actx;i<=shape->rightpix;i++,cmdptr++)
702 {
703 lpix=rpix;
704
705 if(lpix>=viewwidth)
706 break;
707
708 pixcnt+=pixheight;
709 rpix=(pixcnt>>6)+actx;
710
711 if(lpix!=rpix && rpix>0)
712 {
713 if(lpix<0)
714 lpix=0;
715
716 if(rpix>viewwidth)
717 rpix=viewwidth,i=shape->rightpix+1;
718
719 cline=(byte *)shape + *cmdptr;
720
721 while(lpix<rpix)
722 {
723 if(wallheight[lpix]<=(int)height)
724 {
725 line=cline;
726 while((endy = READWORD(line)) != 0)
727 {
728 endy >>= 1;
729 newstart = READWORD(line);
730 starty = READWORD(line) >> 1;
731 j=starty;
732 ycnt=j*pixheight;
733 screndy=(ycnt>>6)+upperedge;
734
735 if(screndy<0)
736 vmem=vbuf+lpix;
737 else
738 vmem=vbuf+screndy*vbufPitch+lpix;
739
740 for(;j<endy;j++)
741 {
742 scrstarty=screndy;
743 ycnt+=pixheight;
744 screndy=(ycnt>>6)+upperedge;
745 if(scrstarty!=screndy && screndy>0)
746 {
747 col=((byte *)shape)[newstart+j];
748
749 if(scrstarty<0)
750 scrstarty=0;
751
752 if(screndy>viewheight)
753 screndy=viewheight,j=endy;
754
755 while(scrstarty<screndy)
756 {
757 *vmem=col;
758 vmem+=vbufPitch;
759 scrstarty++;
760 }
761 }
762 }
763 }
764 }
765 lpix++;
766 }
767 }
768 }
769}

845/*
846=====================
847=
848= DrawScaleds
849=
850= Draws all objects that are visable
851=
852=====================
853*/
854
855#define MAXVISABLE 250
856
857typedef struct
858{
859 short viewx,
860 viewheight,
861 shapenum;
862 short flags; // this must be changed to uint32_t, when you
863 // you need more than 16-flags for drawing
864} visobj_t;
865
866visobj_t vislist[MAXVISABLE];
867visobj_t *visptr,*visstep,*farthest;
868
869void DrawScaleds (void)
870{
871 int i,least,numvisable,height;
872 byte *tilespot,*visspot;
873 unsigned spotloc;
874
875 statobj_t *statptr;
876 objtype *obj;
877
878 visptr = &vislist[0];
879
880//
881// place static objects
882//
883 for (statptr = &statobjlist[0] ; statptr !=laststatobj ; statptr++)
884 {
885 if ((visptr->shapenum = statptr->shapenum) == -1)
886 continue; // object has been deleted
887
888 if (!*statptr->visspot)
889 continue; // not visable
890
891 if (TransformTile (statptr->tilex,statptr->tiley,
892 &visptr->viewx,&visptr->viewheight) && statptr->flags & FL_BONUS)
893 {
894 GetBonus (statptr);
895 if(statptr->shapenum == -1)
896 continue; // object has been taken
897 }
898
899 if (!visptr->viewheight)
900 continue; // to close to the object
901
902 if (visptr < &vislist[MAXVISABLE-1]) // don't let it overflow
903 {
904 visptr->flags = (short) statptr->flags;
905 visptr++;
906 }
907 }
908
909//
910// place active objects
911//
912 for (obj = player->next;obj;obj=obj->next)
913 {
914 if ((visptr->shapenum = obj->state->shapenum)==0)
915 continue; // no shape
916
917 spotloc = (obj->tilex<<mapshift)+obj->tiley; // optimize: keep in struct?
918 visspot = &spotvis[0][0]+spotloc;

929 || ( *(visspot-63) && !*(tilespot-63) )

982 //
