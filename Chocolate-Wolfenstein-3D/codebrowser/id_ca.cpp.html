156{
157 const int handle = open(filename, O_CREAT | O_WRONLY | O_BINARY, 0644);

211static void CAL_HuffExpand(byte *source, byte *dest, int32_t length, huffnode *hufftable)
212{
213 byte *end;
214 huffnode *headptr, *huffptr;
215
216 if(!length || !dest)
217 {
218 Quit("length or dest is null!");
219 return;
220 }
221
222 headptr = hufftable+254; // head node is always node 254
223
224 int written = 0;
225
226 end=dest+length;
227
228 byte val = *source++;
229 byte mask = 1;
230 word nodeval;
231 huffptr = headptr;
232 while(1)
233 {
234 if(!(val & mask))
235 nodeval = huffptr->bit0;
236 else
237 nodeval = huffptr->bit1;
238 if(mask==0x80)
239 {
240 val = *source++;
241 mask = 1;
242 }
243 else mask <<= 1;
244
245 if(nodeval<256)
246 {
247 *dest++ = (byte) nodeval;
248 written++;
249 huffptr = headptr;
250 if(dest>=end) break;
251 }
252 else
253 {
254 huffptr = hufftable + (nodeval - 256);
255 }
256 }
257}

259/*
260======================
261=
262= CAL_CarmackExpand
263=
264= Length is the length of the EXPANDED data
265=
266======================
267*/
268
269#define NEARTAG 0xa7
270#define FARTAG 0xa8
271
272void CAL_CarmackExpand (byte *source, word *dest, int length)
273{
274 word ch,chhigh,count,offset;
275 byte *inptr;
276 word *copyptr, *outptr;
277
278 length/=2;
279
280 inptr = (byte *) source;
281 outptr = dest;
282
283 while (length>0)
284 {
285 ch = READWORD(inptr);
286 chhigh = ch>>8;
287 if (chhigh == NEARTAG)
288 {
289 count = ch&0xff;
290 if (!count)
291 { // have to insert a word containing the tag byte
292 ch |= *inptr++;
293 *outptr++ = ch;
294 length--;
295 }
296 else
297 {
298 offset = *inptr++;
299 copyptr = outptr - offset;
300 length -= count;
301 if(length<0) return;
302 while (count--)
303 *outptr++ = *copyptr++;
304 }
305 }
306 else if (chhigh == FARTAG)
307 {
308 count = ch&0xff;
309 if (!count)
310 { // have to insert a word containing the tag byte
311 ch |= *inptr++;
312 *outptr++ = ch;
313 length --;
314 }
315 else
316 {
317 offset = READWORD(inptr);
318 copyptr = dest + offset;
319 length -= count;
320 if(length<0) return;
321 while (count--)
322 *outptr++ = *copyptr++;
323 }
324 }
325 else
326 {
327 *outptr++ = ch;
328 length --;
329 }
330 }
331}

333/*
334======================
335=
336= CA_RLEWcompress
337=
338======================
339*/
340
341int32_t CA_RLEWCompress (word *source, int32_t length, word *dest, word rlewtag)
342{
343 word value,count;
344 unsigned i;
345 word *start,*end;
346
347 start = dest;
348
349 end = source + (length+1)/2;
350
351 //
352 // compress it
353 //
354 do
355 {
356 count = 1;
357 value = *source++;
358 while (*source == value && source<end)
359 {
360 count++;
361 source++;
362 }
363 if (count>3 || value == rlewtag)
364 {
365 //
366 // send a tag / count / value string
367 //
368 *dest++ = rlewtag;
369 *dest++ = count;
370 *dest++ = value;
371 }
372 else
373 {
374 //
375 // send word without compressing
376 //
377 for (i=1;i<=count;i++)
378 *dest++ = value;
379 }
380
381 } while (source<end);
382
383 return (int32_t)(2*(dest-start));
384}

387/*
388======================
389=
390= CA_RLEWexpand
391= length is EXPANDED length
392=
393======================
394*/
395
396void CA_RLEWexpand (word *source, word *dest, int32_t length, word rlewtag)
397{
398 word value,count,i;
399 word *end=dest+length/2;
400
401//
402// expand it
403//
404 do
405 {
406 value = *source++;
407 if (value != rlewtag)
408 //
409 // uncompressed
410 //
411 *dest++=value;
412 else
413 {
414 //
415 // compressed string
416 //
417 count = *source++;
418 value = *source++;
419 for (i=1;i<=count;i++)
420 *dest++ = value;
421 }
422 } while (dest<end);
423}

443
444void CAL_SetupGrFile (void)
445{
446 char fname[13];
447 int handle;
448 byte *compseg;
449
450#ifdef GRHEADERLINKED
451
452 grhuffman = (huffnode *)&EGAdict;
453 grstarts = (int32_t _seg *)FP_SEG(&EGAhead);
454
455#else
456
457//
458// load ???dict.ext (huffman dictionary for graphics files)
459//
460
461 strcpy(fname,gdictname);
462 strcat(fname,graphext);
463
464 handle = open(fname, O_RDONLY | O_BINARY);
465 if (handle == -1)
466 CA_CannotOpen(fname);
467
468 read(handle, grhuffman, sizeof(grhuffman));
469 close(handle);
470
471 // load the data offsets from ???head.ext
472 strcpy(fname,gheadname);
473 strcat(fname,graphext);
474
475 handle = open(fname, O_RDONLY | O_BINARY);
476 if (handle == -1)
477 CA_CannotOpen(fname);
478
479 long headersize = lseek(handle, 0, SEEK_END);
480 lseek(handle, 0, SEEK_SET);
481
482#ifndef APOGEE_1_0
483 int expectedsize = lengthof(grstarts) - numEpisodesMissing;
484#else
485 int expectedsize = lengthof(grstarts);
486#endif
487
488 if(!param_ignorenumchunks && headersize / 3 != (long) expectedsize)
489 Quit("Chocolate Wolfenstein 3D was not compiled for these data files:\n"
490 "%s contains a wrong number of offsets (%i instead of %i)!\n\n"
491 "Please check whether you are using the right executable!\n"
492 "(For mod developers: perhaps you forgot to update NUMCHUNKS?)",
493 fname, headersize / 3, expectedsize);
494
495 byte data[lengthof(grstarts) * 3];
496 read(handle, data, sizeof(data));
497 close(handle);
498
499 const byte* d = data;
500 for (int32_t* i = grstarts; i != endof(grstarts); ++i)
501 {
502 const int32_t val = d[0] | d[1] << 8 | d[2] << 16;
503 *i = (val == 0x00FFFFFF ? -1 : val);
504 d += 3;
505 }
506#endif
507
508//
509// Open the graphics file, leaving it open until the game is finished
510//
511 strcpy(fname,gfilename);
512 strcat(fname,graphext);
513
514 grhandle = open(fname, O_RDONLY | O_BINARY);
515 if (grhandle == -1)
516 CA_CannotOpen(fname);
517
518
519//
520// load the pic and sprite headers into the arrays in the data segment
521//
522 pictable=(pictabletype *) malloc(NUMPICS*sizeof(pictabletype));
523 CHECKMALLOCRESULT(pictable);
524 CAL_GetGrChunkLength(STRUCTPIC); // position file pointer
525 compseg=(byte *) malloc(chunkcomplen);
526 CHECKMALLOCRESULT(compseg);
527 read (grhandle,compseg,chunkcomplen);
528 CAL_HuffExpand(compseg, (byte*)pictable, NUMPICS * sizeof(pictabletype), grhuffman);
529 free(compseg);
530}

543void CAL_SetupMapFile (void)
544{
545 int i;
546 int handle;
547 int32_t length,pos;
548 char fname[13];
549
550//
551// load maphead.ext (offsets and tileinfo for map file)
552//
553 strcpy(fname,mheadname);
554 strcat(fname,extension);
555
556 handle = open(fname, O_RDONLY | O_BINARY);
557 if (handle == -1)
558 CA_CannotOpen(fname);
559
560 length = NUMMAPS*4+2; // used to be "filelength(handle);"
561 mapfiletype *tinf=(mapfiletype *) malloc(sizeof(mapfiletype));
562 CHECKMALLOCRESULT(tinf);
563 read(handle, tinf, length);
564 close(handle);
565
566 RLEWtag=tinf->RLEWtag;
567
568//
569// open the data file
570//
571#ifdef CARMACIZED
572 strcpy(fname, "gamemaps.");
573 strcat(fname, extension);
574
575 maphandle = open(fname, O_RDONLY | O_BINARY);
576 if (maphandle == -1)
577 CA_CannotOpen(fname);
578#else
579 strcpy(fname,mfilename);
580 strcat(fname,extension);
581
582 maphandle = open(fname, O_RDONLY | O_BINARY);
583 if (maphandle == -1)
584 CA_CannotOpen(fname);
585#endif
586
587//
588// load all map header
589//
590 for (i=0;i<NUMMAPS;i++)
591 {
592 pos = tinf->headeroffsets[i];
593 if (pos<0) // $FFFFFFFF start is a sparse map
594 continue;
595
596 mapheaderseg[i]=(maptype *) malloc(sizeof(maptype));
597 CHECKMALLOCRESULT(mapheaderseg[i]);
598 lseek(maphandle,pos,SEEK_SET);
599 read (maphandle,(memptr)mapheaderseg[i],sizeof(maptype));
600 }
601
602 free(tinf);
603
604//
605// allocate space for 3 64*64 planes
606//
607 for (i=0;i<MAPPLANES;i++)
608 {
609 mapsegs[i]=(word *) malloc(maparea*2);
610 CHECKMALLOCRESULT(mapsegs[i]);
611 }
612}

750void CA_CacheAdlibSoundChunk (int chunk)
751{
752 int32_t pos = audiostarts[chunk];
753 int32_t size = audiostarts[chunk+1]-pos;
754
755 if (audiosegs[chunk])
756 return; // already in memory
757
758 lseek(audiohandle, pos, SEEK_SET);
759 read(audiohandle, bufferseg, ORIG_ADLIBSOUND_SIZE - 1); // without data[1]
760
761 AdLibSound *sound = (AdLibSound *) malloc(size + sizeof(AdLibSound) - ORIG_ADLIBSOUND_SIZE);
762 CHECKMALLOCRESULT(sound);
763
764 byte *ptr = (byte *) bufferseg;
765 sound->common.length = READLONGWORD(ptr);
766 sound->common.priority = READWORD(ptr);
767 sound->inst.mChar = *ptr++;
768 sound->inst.cChar = *ptr++;
769 sound->inst.mScale = *ptr++;
770 sound->inst.cScale = *ptr++;
771 sound->inst.mAttack = *ptr++;
772 sound->inst.cAttack = *ptr++;
773 sound->inst.mSus = *ptr++;
774 sound->inst.cSus = *ptr++;
775 sound->inst.mWave = *ptr++;
776 sound->inst.cWave = *ptr++;
777 sound->inst.nConn = *ptr++;
778 sound->inst.voice = *ptr++;
779 sound->inst.mode = *ptr++;
780 sound->inst.unused[0] = *ptr++;
781 sound->inst.unused[1] = *ptr++;
782 sound->inst.unused[2] = *ptr++;
783 sound->block = *ptr++;
784
785 read(audiohandle, sound->data, size - ORIG_ADLIBSOUND_SIZE + 1); // + 1 because of byte data[1]
786
787 audiosegs[chunk]=(byte *) sound;
788}

793======================
794=
795= CA_LoadAllSounds
796=
797= Purges all sounds, then loads all new ones (mode switch)
798=
799======================
800*/
801
802void CA_LoadAllSounds (void)
803{
804 unsigned start,i;
805
806 switch (oldsoundmode)
807 {
808 case sdm_Off:
809 goto cachein;
810 case sdm_PC:
811 start = STARTPCSOUNDS;
812 break;
813 case sdm_AdLib:
814 start = STARTADLIBSOUNDS;
815 break;
816 }
817
818 for (i=0;i<NUMSOUNDS;i++,start++)
819 UNCACHEAUDIOCHUNK(start);
820
821cachein:
822
823 oldsoundmode = SoundMode;
824
825 switch (SoundMode)
826 {
827 case sdm_Off:
828 start = STARTADLIBSOUNDS; // needed for priorities...
829 break;
830 case sdm_PC:
831 start = STARTPCSOUNDS;
832 break;
833 case sdm_AdLib:
834 start = STARTADLIBSOUNDS;
835 break;
836 }
837
838 if(start == STARTADLIBSOUNDS)
839 {
840 for (i=0;i<NUMSOUNDS;i++,start++)
841 CA_CacheAdlibSoundChunk(start);
842 }
843 else
844 {
845 for (i=0;i<NUMSOUNDS;i++,start++)
846 CA_CacheAudioChunk(start);
847 }
848}

853/*
854======================
855=
856= CAL_ExpandGrChunk
857=
858= Does whatever is needed with a pointer to a compressed chunk
859=
860======================
861*/
862
863void CAL_ExpandGrChunk (int chunk, int32_t *source)
864{
865 int32_t expanded;
866
867 if (chunk >= STARTTILE8 && chunk < STARTEXTERNS)
868 {
869 //
870 // expanded sizes of tile8/16/32 are implicit
871 //
872
873#define BLOCK 64
874#define MASKBLOCK 128
875
876 if (chunk<STARTTILE8M) // tile 8s are all in one chunk!
877 expanded = BLOCK*NUMTILE8;
878 else if (chunk<STARTTILE16)
879 expanded = MASKBLOCK*NUMTILE8M;
880 else if (chunk<STARTTILE16M) // all other tiles are one/chunk
881 expanded = BLOCK*4;
882 else if (chunk<STARTTILE32)
883 expanded = MASKBLOCK*4;
884 else if (chunk<STARTTILE32M)
885 expanded = BLOCK*16;
886 else
887 expanded = MASKBLOCK*16;
888 }
889 else
890 {
891 //
892 // everything else has an explicit size longword
893 //
894 expanded = *source++;
895 }
896
897 //
898 // allocate final space, decompress it, and free bigbuffer
899 // Sprites need to have shifts made and various other junk
900 //
901 grsegs[chunk]=(byte *) malloc(expanded);
902 CHECKMALLOCRESULT(grsegs[chunk]);
903 CAL_HuffExpand((byte *) source, grsegs[chunk], expanded, grhuffman);
904}

907/*
908======================
909=
910= CA_CacheGrChunk
911=
912= Makes sure a given chunk is in memory, loadiing it if needed
913=
914======================
915*/
916
917void CA_CacheGrChunk (int chunk)
918{
919 int32_t pos,compressed;
920 int32_t *source;
921 int next;
922
923 if (grsegs[chunk])
924 return; // already in memory
925
926//
927// load the chunk into a buffer, either the miscbuffer if it fits, or allocate
928// a larger buffer
929//
930 pos = GRFILEPOS(chunk);
931 if (pos<0) // $FFFFFFFF start is a sparse tile
932 return;
933
934 next = chunk +1;
935 while (GRFILEPOS(next) == -1) // skip past any sparse tiles
936 next++;
937
938 compressed = GRFILEPOS(next)-pos;
939
940 lseek(grhandle,pos,SEEK_SET);
941
942 if (compressed<=BUFFERSIZE)
943 {
944 read(grhandle,bufferseg,compressed);
945 source = bufferseg;
946 }
947 else
948 {
949 source = (int32_t *) malloc(compressed);
950 CHECKMALLOCRESULT(source);
951 read(grhandle,source,compressed);
952 }
953
954 CAL_ExpandGrChunk (chunk,source);
955
956 if (compressed>BUFFERSIZE)
957 free(source);
958}

964/*
965======================
966=
967= CA_CacheScreen
968=
969= Decompresses a chunk from disk straight onto the screen
970=
971======================
972*/
973
974void CA_CacheScreen (int chunk)
975{
976 int32_t pos,compressed,expanded;
977 memptr bigbufferseg;
978 int32_t *source;
979 int next;
980
981//
982// load the chunk into a buffer
983//
984 pos = GRFILEPOS(chunk);
985 next = chunk +1;
986 while (GRFILEPOS(next) == -1) // skip past any sparse tiles
987 next++;
988 compressed = GRFILEPOS(next)-pos;
989
990 lseek(grhandle,pos,SEEK_SET);
991
992 bigbufferseg=malloc(compressed);
993 CHECKMALLOCRESULT(bigbufferseg);
994 read(grhandle,bigbufferseg,compressed);
995 source = (int32_t *) bigbufferseg;
996
997 expanded = *source++;
998
999//
1000// allocate final space, decompress it, and free bigbuffer
1001// Sprites need to have shifts made and various other junk
1002//
1003 byte *pic = (byte *) malloc(64000);
1004 CHECKMALLOCRESULT(pic);
1005 CAL_HuffExpand((byte *) source, pic, expanded, grhuffman);
1006
1007 byte *vbuf = LOCK();
1008 for(int y = 0, scy = 0; y < 200; y++, scy += scaleFactor)
1009 {
1010 for(int x = 0, scx = 0; x < 320; x++, scx += scaleFactor)
1011 {
1012 byte col = pic[(y * 80 + (x >> 2)) + (x & 3) * 80 * 200];
1013 for(unsigned i = 0; i < scaleFactor; i++)
1014 for(unsigned j = 0; j < scaleFactor; j++)
1015 vbuf[(scy + i) * curPitch + scx + j] = col;
1016 }
1017 }
1018 UNLOCK();
1019 free(pic);
1020 free(bigbufferseg);
1021}

1035void CA_CacheMap (int mapnum)
1036{
1037 int32_t pos,compressed;
1038 int plane;
1039 word *dest;
1040 memptr bigbufferseg;
1041 unsigned size;
1042 word *source;
1043#ifdef CARMACIZED
1044 word *buffer2seg;
1045 int32_t expanded;
1046#endif
1047
1048 mapon = mapnum;
1049
1050//
1051// load the planes into the allready allocated buffers
1052//
1053 size = maparea*2;
1054
1055 for (plane = 0; plane<MAPPLANES; plane++)
1056 {
1057 pos = mapheaderseg[mapnum]->planestart[plane];
1058 compressed = mapheaderseg[mapnum]->planelength[plane];
1059
1060 dest = mapsegs[plane];
1061
1062 lseek(maphandle,pos,SEEK_SET);
1063 if (compressed<=BUFFERSIZE)
1064 source = (word *) bufferseg;
1065 else
1066 {
1067 bigbufferseg=malloc(compressed);
1068 CHECKMALLOCRESULT(bigbufferseg);
1069 source = (word *) bigbufferseg;
1070 }
1071
1072 read(maphandle,source,compressed);
1073#ifdef CARMACIZED
1074 //
1075 // unhuffman, then unRLEW
1076 // The huffman'd chunk has a two byte expanded length first
1077 // The resulting RLEW chunk also does, even though it's not really
1078 // needed
1079 //
1080 expanded = *source;
1081 source++;
1082 buffer2seg = (word *) malloc(expanded);
1083 CHECKMALLOCRESULT(buffer2seg);
1084 CAL_CarmackExpand((byte *) source, buffer2seg,expanded);
1085 CA_RLEWexpand(buffer2seg+1,dest,size,RLEWtag);
1086 free(buffer2seg);
1087
1088#else
1089 //
1090 // unRLEW, skipping expanded length
1091 //
1092 CA_RLEWexpand (source+1,dest,size,RLEWtag);
1093#endif
1094
1095 if (compressed>BUFFERSIZE)
1096 free(bigbufferseg);
1097 }
1098}
