358void PollJoystickMove (void)
359{
360 int joyx, joyy;
361
362 IN_GetJoyDelta (&joyx, &joyy);
363
364 int delta = buttonstate[bt_run] ? RUNMOVE * tics : BASEMOVE * tics;
365
366 if (joyx > 64 || buttonstate[bt_turnright])
367 controlx += delta;
368 else if (joyx < -64 || buttonstate[bt_turnleft])
369 controlx -= delta;
370 if (joyy > 64 || buttonstate[bt_movebackward])
371 controly += delta;
372 else if (joyy < -64 || buttonstate[bt_moveforward])
373 controly -= delta;
374}

376/*
377===================
378=
379= PollControls
380=
381= Gets user or demo input, call once each frame
382=
383= controlx set between -100 and 100 per tic
384= controly
385= buttonheld[] the state of the buttons LAST frame
386= buttonstate[] the state of the buttons THIS frame
387=
388===================
389*/
390
391void PollControls (void)
392{
393 int max, min, i;
394 byte buttonbits;
395
396 IN_ProcessEvents();
397
398//
399// get timing info for last frame
400//
401 if (demoplayback || demorecord) // demo recording and playback needs to be constant
402 {
403 // wait up to DEMOTICS Wolf tics
404 uint32_t curtime = SDL_GetTicks();
405 lasttimecount += DEMOTICS;
406 int32_t timediff = (lasttimecount * 100) / 7 - curtime;
407 if(timediff > 0)
408 SDL_Delay(timediff);
409
410 if(timediff < -2 * DEMOTICS) // more than 2-times DEMOTICS behind?
411 lasttimecount = (curtime * 7) / 100; // yes, set to current timecount
412
413 tics = DEMOTICS;
414 }
415 else
416 CalcTics ();
417
418 controlx = 0;
419 controly = 0;
420 memcpy (buttonheld, buttonstate, sizeof (buttonstate));
421 memset (buttonstate, 0, sizeof (buttonstate));
422
423 if (demoplayback)
424 {
425 //
426 // read commands from demo buffer
427 //
428 buttonbits = *demoptr++;
429 for (i = 0; i < NUMBUTTONS; i++)
430 {
431 buttonstate[i] = buttonbits & 1;
432 buttonbits >>= 1;
433 }
434
435 controlx = *demoptr++;
436 controly = *demoptr++;
437
438 if (demoptr == lastdemoptr)
439 playstate = ex_completed; // demo is done
440
441 controlx *= (int) tics;
442 controly *= (int) tics;
443
444 return;
445 }
446
447
448//
449// get button states
450//
451 PollKeyboardButtons ();
452
453 if (mouseenabled && IN_IsInputGrabbed())
454 PollMouseButtons ();
455
456 if (joystickenabled)
457 PollJoystickButtons ();
458
459//
460// get movements
461//
462 PollKeyboardMove ();
463
464 if (mouseenabled && IN_IsInputGrabbed())
465 PollMouseMove ();
466
467 if (joystickenabled)
468 PollJoystickMove ();
469
470//
471// bound movement to a maximum
472//
473 max = 100 * tics;
474 min = -max;
475 if (controlx > max)
476 controlx = max;
477 else if (controlx < min)
478 controlx = min;
479
480 if (controly > max)
481 controly = max;
482 else if (controly < min)
483 controly = min;
484
485 if (demorecord)
486 {
487 //
488 // save info out to demo buffer
489 //
490 controlx /= (int) tics;
491 controly /= (int) tics;
492
493 buttonbits = 0;
494
495 // TODO: Support 32-bit buttonbits
496 for (i = NUMBUTTONS - 1; i >= 0; i--)
497 {
498 buttonbits <<= 1;
499 if (buttonstate[i])
500 buttonbits |= 1;
501 }
502
503 *demoptr++ = buttonbits;
504 *demoptr++ = controlx;
505 *demoptr++ = controly;
506
507 if (demoptr >= lastdemoptr - 8)
508 playstate = ex_completed;
509 else
510 {
511 controlx *= (int) tics;
512 controly *= (int) tics;
513 }
514 }
515}

744/*
745#############################################################################
746
747 The objlist data structure
748
749#############################################################################
750
751objlist containt structures for every actor currently playing. The structure
752is accessed as a linked list starting at *player, ending when ob->next ==
753NULL. GetNewObj inserts a new object at the end of the list, meaning that
754if an actor spawn another actor, the new one WILL get to think and react the
755same frame. RemoveObj unlinks the given object and returns it to the free
756list, but does not damage the objects ->next pointer, so if the current object
757removes itself, a linked list following loop can still safely get to the
758next element.
759
760<backwardly linked free list>
761
762#############################################################################
763*/
764
765
766/*
767=========================
768=
769= InitActorList
770=
771= Call to clear out the actor object lists returning them all to the free
772= list. Allocates a special spot for the player.
773=
774=========================
775*/
776
777int objcount;
778
779void InitActorList (void)
780{
781 int i;
782
783//
784// init the actor lists
785//
786 for (i = 0; i < MAXACTORS; i++)
787 {
788 objlist[i].prev = &objlist[i + 1];
789 objlist[i].next = NULL;
790 }
791
792 objlist[MAXACTORS - 1].prev = NULL;
793
794 objfreelist = &objlist[0];
795 lastobj = NULL;
796
797 objcount = 0;
798
799//
800// give the player the first free spots
801//
802 GetNewActor ();
803 player = newobj;
804
805}
806
807//===========================================================================
808
809/*
810=========================
811=
812= GetNewActor
813=
814= Sets the global variable new to point to a free spot in objlist.
815= The free spot is inserted at the end of the liked list
816=
817= When the object list is full, the caller can either have it bomb out ot
818= return a dummy object pointer that will never get used
819=
820=========================
821*/
822
823void GetNewActor (void)
824{
825 if (!objfreelist)
826 Quit ("GetNewActor: No free spots in objlist!");
827
828 newobj = objfreelist;
829 objfreelist = newobj->prev;
830 memset (newobj, 0, sizeof (*newobj));
831
832 if (lastobj)
833 lastobj->next = newobj;
834 newobj->prev = lastobj; // new->next is allready NULL from memset
835
836 newobj->active = ac_no;
837 lastobj = newobj;
838
839 objcount++;
840}
841
842//===========================================================================
843
844/*
845=========================
846=
847= RemoveObj
848=
849= Add the given object back into the free list, and unlink it from it's
850= neighbors
851=
852=========================
853*/
854
855void RemoveObj (objtype * gone)
856{
857 if (gone == player)
858 Quit ("RemoveObj: Tried to remove the player!");
859
860 gone->state = NULL;
861
862//
863// fix the next object's back link
864//
865 if (gone == lastobj)
866 lastobj = (objtype *) gone->prev;
867 else
868 gone->next->prev = gone->prev;
869
870//
871// fix the previous object's forward link
872//
873 gone->prev->next = gone->next;
874
875//
876// add it back in to the free list
877//
878 gone->prev = objfreelist;
879 objfreelist = gone;
880
881 objcount--;
882}

956/*
957=====================
958=
959= InitRedShifts
960=
961=====================
962*/
963
964void InitRedShifts (void)
965{
966 SDL_Color *workptr, *baseptr;
967 int i, j, delta;
968
969
970//
971// fade through intermediate frames
972//
973 for (i = 1; i <= NUMREDSHIFTS; i++)
974 {
975 workptr = redshifts[i - 1];
976 baseptr = gamepal;
977
978 for (j = 0; j <= 255; j++)
979 {
980 delta = 256 - baseptr->r;
981 workptr->r = baseptr->r + delta * i / REDSTEPS;
982 delta = -baseptr->g;
983 workptr->g = baseptr->g + delta * i / REDSTEPS;
984 delta = -baseptr->b;
985 workptr->b = baseptr->b + delta * i / REDSTEPS;
986 baseptr++;
987 workptr++;
988 }
989 }
990
991 for (i = 1; i <= NUMWHITESHIFTS; i++)
992 {

1004 baseptr++;

1147void DoActor (objtype * ob)
1148{
1149 void (*think) (objtype *);
1150
1151 if (!ob->active && !areabyplayer[ob->areanumber])
1152 return;
1153
1154 if (!(ob->flags & (FL_NONMARK | FL_NEVERMARK)))
1155 actorat[ob->tilex][ob->tiley] = NULL;
1156
1157//
1158// non transitional object
1159//
1160
1161 if (!ob->ticcount)
1162 {
1163 think = (void (*)(objtype *)) ob->state->think;
1164 if (think)
1165 {
1166 think (ob);
1167 if (!ob->state)
1168 {
1169 RemoveObj (ob);
1170 return;
1171 }
1172 }
1173
1174 if (ob->flags & FL_NEVERMARK)
1175 return;
1176
1177 if ((ob->flags & FL_NONMARK) && actorat[ob->tilex][ob->tiley])
1178 return;
1179
1180 actorat[ob->tilex][ob->tiley] = ob;
1181 return;
1182 }
1183
1184//
1185// transitional object
1186//
1187 ob->ticcount -= (short) tics;
1188 while (ob->ticcount <= 0)
1189 {
1190 think = (void (*)(objtype *)) ob->state->action; // end of state action
1191 if (think)
1192 {
1193 think (ob);
1194 if (!ob->state)
1195 {
1196 RemoveObj (ob);
1197 return;
1198 }
1199 }
1200
1201 ob->state = ob->state->next;
1202
1203 if (!ob->state)
1204 {
1205 RemoveObj (ob);
1206 return;
1207 }
1208
1209 if (!ob->state->tictime)
1210 {
1211 ob->ticcount = 0;
1212 goto think;
1213 }
1214
1215 ob->ticcount += ob->state->tictime;
1216 }
1217
1218think:
1219 //
1220 // think
1221 //
1222 think = (void (*)(objtype *)) ob->state->think;
1223 if (think)
1224 {
1225 think (ob);
1226 if (!ob->state)
1227 {
1228 RemoveObj (ob);
1229 return;
1230 }
1231 }
1232
1233 if (ob->flags & FL_NEVERMARK)
1234 return;
1235
1236 if ((ob->flags & FL_NONMARK) && actorat[ob->tilex][ob->tiley])
1237 return;
1238
1239 actorat[ob->tilex][ob->tiley] = ob;
1240}
