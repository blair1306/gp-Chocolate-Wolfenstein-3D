74
75 US_Print ("Total statics :");

78
79 char str[60];
80 sprintf(str,"\nlaststatobj=%.8X",(int32_t)(uintptr_t)laststatobj);
81 US_Print(str);
82
83 US_Print ("\nIn use statics:");
84 for (i=0;i<total;i++)
85 {
86 if (statobjlist[i].shapenum != -1)
87 count++;
88 else
89 doors++; //debug
90 }
91 US_PrintUnsigned (count);
92
93 US_Print ("\nDoors :");
94 US_PrintUnsigned (doornum);
95
96 for (obj=player->next;obj;obj=obj->next)
97 {
98 if (obj->active)
99 active++;
100 else
101 inactive++;
102 }
103
104 US_Print ("\nTotal actors :");
105 US_PrintUnsigned (active+inactive);
106
107 US_Print ("\nActive actors :");
108 US_PrintUnsigned (active);
109

124void PictureGrabber (void)
125{
126 static char fname[] = "WSHOT000.BMP";
127
128 for(int i = 0; i < 1000; i++)
129 {
130 fname[7] = i % 10 + '0';
131 fname[6] = (i / 10) % 10 + '0';
132 fname[5] = i / 100 + '0';
133 int file = open(fname, O_RDONLY | O_BINARY);
134 if(file == -1) break; // file does not exist, so use that filename
135 close(file);
136 }
137
138 // overwrites WSHOT999.BMP if all wshot files exist
139
140 SDL_SaveBMP(curSurface, fname);
141
142 CenterWindow (18,2);
143 US_PrintCentered ("Screenshot taken");
144 VW_UpdateScreen();
145 IN_Ack();
146}

159void BasicOverhead (void)
160{
161 int x, y, z, offx, offy;
162
163 z = 128/MAPSIZE; // zoom scale
164 offx = 320/2;
165 offy = (160-MAPSIZE*z)/2;
166
167#ifdef MAPBORDER
168 int temp = viewsize;
169 NewViewSize(16);
170 DrawPlayBorder();
171#endif
172
173 // right side (raw)
174
175 for(x=0;x<MAPSIZE;x++)
176 for(y=0;y<MAPSIZE;y++)
177 VWB_Bar(x*z+offx, y*z+offy,z,z,(unsigned)(uintptr_t)actorat[x][y]);
178
179 // left side (filtered)
180
181 uintptr_t tile;
182 int color;
183 offx -= 128;
184
185 for(x=0;x<MAPSIZE;x++)
186 {
187 for(y=0;y<MAPSIZE;y++)
188 {
189 tile = (uintptr_t)actorat[x][y];
190 if (ISPOINTER(tile) && ((objtype *)tile)->flags&FL_SHOOTABLE) color = 72; // enemy
191 else if (!tile || ISPOINTER(tile))
192 {
193 if (spotvis[x][y]) color = 111; // visable
194 else color = 0; // nothing
195 }
196 else if (MAPSPOT(x,y,1) == PUSHABLETILE) color = 171; // pushwall
197 else if (tile == 64) color = 158; // solid obj
198 else if (tile < 128) color = 154; // walls
199 else if (tile < 256) color = 146; // doors
200
201 VWB_Bar(x*z+offx, y*z+offy,z,z,color);
202 }
203 }
204
205 VWB_Bar(player->tilex*z+offx,player->tiley*z+offy,z,z,15); // player
206
207 // resize the border to match
208
209 VW_UpdateScreen();
210 IN_Ack();
211
212#ifdef MAPBORDER
213 NewViewSize(temp);
214 DrawPlayBorder();
215#endif
216}
